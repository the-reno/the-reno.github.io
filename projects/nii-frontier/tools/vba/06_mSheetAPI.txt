Attribute VB_Name = "mSheetAPI"
' =====================================================================
' mSheetAPI - the worksheet functions (the only module that reads cells).
' Build objects with FED_HOL and MAKE_CURVE; they return a short receipt
' and stash the object by name. Then ACCRUE and SWAP take the receipt
' CELL (e.g. $B$5) and do the math. Never type the receipt text by hand.
' =====================================================================
Option Explicit

' =FED_HOL(holidays)  ->  "hol_calendar | OK | 6 dates"
'   holidays: a one-column range of dates. Weekends are automatic.
Public Function FED_HOL(holidays As Range) As String
    On Error GoTo bad
    Dim cal As New cCalendar
    cal.Init holidays.Value
    RegSet "hol_calendar", cal
    FED_HOL = "hol_calendar | OK | " & cal.HolidayCount & " dates"
    Exit Function
bad:
    FED_HOL = "#CAL_ERR: " & Err.Description
End Function

' =MAKE_CURVE(name, sofr, fedMoves, calendar)
'   name     : text, e.g. "cuts"
'   sofr     : today's SOFR %, e.g. 3.80
'   fedMoves : a TWO-COLUMN range -> [meeting date | move bps]. Blank rows
'              ignored. Each move takes effect the next business day.
'   calendar : the FED_HOL receipt cell.
'   -> "cuts | OK | 3 steps | 3.80->3.05"
Public Function MAKE_CURVE(ByVal name As String, ByVal sofr As Double, _
                           fedMoves As Range, calendar As Range) As String
    On Error GoTo bad
    Dim cal As cCalendar
    Set cal = RegGet(HandleName(CStr(calendar.Value)))
    If cal Is Nothing Then MAKE_CURVE = "#CURVE_ERR: build the calendar first": Exit Function

    Dim n As Long, i As Long, fd() As Variant, mv() As Variant
    n = fedMoves.Rows.Count
    ReDim fd(1 To n): ReDim mv(1 To n)
    For i = 1 To n
        fd(i) = fedMoves.Cells(i, 1).Value
        mv(i) = fedMoves.Cells(i, 2).Value
    Next i

    Dim crv As New cStripCurve
    crv.Init name, sofr, fd, mv, cal
    RegSet name, crv
    MAKE_CURVE = name & " | OK | " & crv.Steps & " steps | " & _
                 Format(sofr, "0.00") & "->" & Format(crv.LastRate, "0.00")
    Exit Function
bad:
    MAKE_CURVE = "#CURVE_ERR: " & Err.Description
End Function

' =ACCRUE(start, end, amount, type, curve)
'   type : "SIMPLE" (sum of daily interest) or "COMPOUND" (interest joins
'          the balance, SOFR in-arrears - the market floating-leg method).
'   curve: the MAKE_CURVE receipt cell.
' Returns interest in $mm for [start, end).
Public Function ACCRUE(ByVal startDate As Date, ByVal endDate As Date, _
                       ByVal amount As Double, ByVal accrualType As String, _
                       curve As Range) As Variant
    On Error GoTo bad
    Dim crv As cStripCurve
    Set crv = RegGet(HandleName(CStr(curve.Value)))
    If crv Is Nothing Then ACCRUE = CVErr(xlErrNA): Exit Function
    Select Case UCase$(accrualType)
        Case "SIMPLE":   ACCRUE = amount * crv.SimpleFactor(startDate, endDate)
        Case "COMPOUND": ACCRUE = amount * (crv.AccumFactor(startDate, endDate) - 1#)
        Case Else:       ACCRUE = CVErr(xlErrValue)
    End Select
    Exit Function
bad:
    ACCRUE = CVErr(xlErrValue)
End Function

' =SWAP(start, end, notional, fixedRate, curve, leg)
'   leg : "FIXED", "FLOAT" or "NET" (NET = receive-fixed: fixed - float).
'   Fixed leg = simple at fixedRate; float leg = COMPOUND on the curve.
Public Function SWAP(ByVal startDate As Date, ByVal endDate As Date, _
                     ByVal notional As Double, ByVal fixedRate As Double, _
                     curve As Range, Optional ByVal leg As String = "NET") As Variant
    On Error GoTo bad
    Dim crv As cStripCurve
    Set crv = RegGet(HandleName(CStr(curve.Value)))
    If crv Is Nothing Then SWAP = CVErr(xlErrNA): Exit Function
    Dim s As Date, fixedLeg As Double, floatLeg As Double
    s = crv.Calendar.Following(startDate)
    fixedLeg = notional * fixedRate / 100# * CLng(endDate - s) / 360#
    floatLeg = notional * (crv.AccumFactor(startDate, endDate) - 1#)
    Select Case UCase$(leg)
        Case "FIXED": SWAP = fixedLeg
        Case "FLOAT": SWAP = floatLeg
        Case "NET":   SWAP = fixedLeg - floatLeg
        Case Else:    SWAP = CVErr(xlErrValue)
    End Select
    Exit Function
bad:
    SWAP = CVErr(xlErrValue)
End Function
