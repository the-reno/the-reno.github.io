Attribute VB_Name = "mSheetAPI"
' =====================================================================
' mSheetAPI (Layer 7, sheet door) - the only module that touches cells.
' Constructors build objects from sheet VALUES, register them, return a
' receipt string. Consumers take handle CELLS (never typed strings, C1)
' and call the engine. Bad input -> "#..._ERR:" receipt at the cell.
' =====================================================================
Option Explicit

' =FED_HOL(holidayRange) -> "hol_calendar | OK | n holidays"
Public Function FED_HOL(holidayRange As Range) As String
    On Error GoTo bad
    Dim cal As New cCalendar
    cal.Init holidayRange.Value
    RegSet "hol_calendar", cal
    FED_HOL = "hol_calendar | OK | " & cal.HolidayCount & " holidays"
    Exit Function
bad:
    FED_HOL = "#CAL_ERR: " & Err.Description
End Function

' =MAKE_CURVE(name, sofr0, fomcDates, movesBps, scale, calHandleCell)
'   -> "name | OK | k steps | 3.80->3.05 | in-arrears ACT/360"
' Each non-zero move is effective the NEXT BUSINESS DAY; scale multiplies.
Public Function MAKE_CURVE(ByVal name As String, ByVal sofr0 As Double, _
                           fomcDates As Range, movesBps As Range, _
                           ByVal scale As Double, calHandle As Range) As String
    On Error GoTo bad
    Dim cal As cCalendar
    Set cal = RegGet(HandleName(CStr(calHandle.Value)))
    If cal Is Nothing Then MAKE_CURVE = "#CURVE_ERR: calendar not built (Ctrl+Alt+F9)": Exit Function
    Dim fd() As Variant, mv() As Variant, i As Long, k As Long
    k = fomcDates.Rows.Count
    ReDim fd(1 To k): ReDim mv(1 To k)
    For i = 1 To k
        fd(i) = fomcDates.Cells(i, 1).Value
        mv(i) = movesBps.Cells(i, 1).Value
    Next i
    Dim crv As New cStripCurve
    crv.Init name, sofr0, fd, mv, scale, cal
    RegSet name, crv
    MAKE_CURVE = name & " | OK | " & crv.Steps & " steps | " & _
                 Format(sofr0, "0.00") & "->" & Format(crv.LastRate, "0.00") & _
                 " | in-arrears ACT/360"
    Exit Function
bad:
    MAKE_CURVE = "#CURVE_ERR: " & Err.Description
End Function

' =ACCRUE(start, end, notionalMM, type, curveHandleCell, info)
'   type = "SIMPLE" | "COMPOUND"   info = "I" | "PI"
' SIMPLE   : notional * SimpleFactor(start,end)
' COMPOUND : notional * (AccumFactor(start,end) - 1)   [SOFR in-arrears]
' info "PI": adds notional back (principal + interest).
' This single function is the deposit, the swap float leg, and (with a
' flat curve) the swap fixed leg.
Public Function ACCRUE(ByVal dStart As Date, ByVal dEnd As Date, _
                       ByVal notionalMM As Double, ByVal accrualType As String, _
                       curveHandle As Range, Optional ByVal info As String = "I") As Variant
    On Error GoTo bad
    Dim crv As cStripCurve
    Set crv = RegGet(HandleName(CStr(curveHandle.Value)))
    If crv Is Nothing Then ACCRUE = CVErr(xlErrNA): Exit Function
    Dim intr As Double
    Select Case UCase$(accrualType)
        Case "SIMPLE":   intr = notionalMM * crv.SimpleFactor(dStart, dEnd)
        Case "COMPOUND": intr = notionalMM * (crv.AccumFactor(dStart, dEnd) - 1#)
        Case Else:       ACCRUE = CVErr(xlErrValue): Exit Function
    End Select
    If UCase$(info) = "PI" Then intr = intr + notionalMM
    ACCRUE = intr
    Exit Function
bad:
    ACCRUE = CVErr(xlErrValue)
End Function

' =SWAP_NET(start, end, notionalMM, fixedPct, curveHandleCell, dir)
' Coordinator: fixed leg = simple at fixed rate; float leg = ACCRUE
' COMPOUND on the scenario curve (market in-arrears); net by direction.
Public Function SWAP_NET(ByVal dStart As Date, ByVal dEnd As Date, _
                         ByVal notionalMM As Double, ByVal fixedPct As Double, _
                         curveHandle As Range, ByVal direction As String) As Variant
    On Error GoTo bad
    Dim crv As cStripCurve
    Set crv = RegGet(HandleName(CStr(curveHandle.Value)))
    If crv Is Nothing Then SWAP_NET = CVErr(xlErrNA): Exit Function
    Dim s As Date, fixedLeg As Double, floatLeg As Double
    s = crv.Calendar.Following(dStart)
    fixedLeg = notionalMM * fixedPct / 100# * CLng(dEnd - s) / 360#
    floatLeg = notionalMM * (crv.AccumFactor(dStart, dEnd) - 1#)
    SWAP_NET = IIf(UCase$(direction) = "RCV", 1#, -1#) * (fixedLeg - floatLeg)
    Exit Function
bad:
    SWAP_NET = CVErr(xlErrValue)
End Function
