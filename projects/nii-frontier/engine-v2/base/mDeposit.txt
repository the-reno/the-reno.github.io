' =====================================================================
' mDeposit  -  deposit daily accrual skeleton
'
' RUN: Alt+F8 > BuildDepositLedger
'
' One self-contained row per business day per active bucket.
' Every column needed to verify the interest independently is present.
' Rate% and Interest($mm) are LEFT BLANK - wire RCurveRate and RAccrue
' to those columns after connecting a curve.
'
' ROW SCHEMA (fully self-contained - auditable without other rows)
'   StartDate   start of the accrual period
'   EndDate     end of the accrual period  (= next business day for ON;
'               = StartDate + 1 day for term buckets daily accrual)
'   Bucket      ON / 1M / 2M / 3M
'   Tenor(days) EndDate - StartDate  (explicit, no calculation needed)
'   RateDate    date the rate is observed from the curve:
'                 ON  -> StartDate  (overnight fixes each morning)
'                 1M/2M/3M -> placement date of the current period
'                             (forward rate locked at inception)
'   Rate%       BLANK -> =RCurveRate(curveCell, E2)
'   Notional    the deposit amount ($mm)
'   Interest    BLANK -> =RAccrue(A2, B2, G2, "SIMPLE", curveCell)
'               or simply: =G2 * F2/100 * D2/360
'
' INPUT CELL MAP (INPUTS sheet)
'   B4  Start date
'   B5  Horizon (years: 0.5 / 1 / 1.5 / 2)
'   B6  ON allocation  ($mm)
'   B7  1M allocation  ($mm)
'   B8  2M allocation  ($mm)
'   B9  3M allocation  ($mm)
'   A14:A30  Holiday dates (one per row, blank stops reading)
' =====================================================================
Option Explicit

Private Hol As Object

Private Sub LoadHolidays(ws As Worksheet)
    Set Hol = CreateObject("Scripting.Dictionary")
    Dim r As Long, v As Variant
    For r = 14 To 50
        v = ws.Cells(r, 1).Value
        If v = "" Then Exit For
        If IsDate(v) Then
            Dim k As Long: k = CLng(CDate(v))
            If Not Hol.Exists(k) Then Hol.Add k, True
        End If
    Next r
End Sub

Private Function IsBusinessDay(d As Date) As Boolean
    Dim w As Long: w = Weekday(d, vbMonday)
    IsBusinessDay = (w <= 5) And Not Hol.Exists(CLng(d))
End Function

Private Function Following(d As Date) As Date
    Do While Not IsBusinessDay(d): d = d + 1: Loop
    Following = d
End Function

Private Function NextBD(d As Date) As Date
    NextBD = Following(d + 1)
End Function

Private Function AddMonths(d As Date, m As Long) As Date
    Dim y As Long, mo As Long, dy As Long
    y = Year(d): mo = Month(d): dy = Day(d)
    mo = mo + m
    Do While mo > 12: mo = mo - 12: y = y + 1: Loop
    Dim last As Long: last = Day(DateSerial(y, mo + 1, 0))
    If dy > last Then dy = last
    AddMonths = DateSerial(y, mo, dy)
End Function

Private Sub WriteRow(led As Worksheet, r As Long, _
                     startD As Date, endD As Date, _
                     bucket As String, tenor As Long, _
                     rateDate As Date, notional As Double)
    led.Cells(r, 1).Value = startD      ' StartDate
    led.Cells(r, 2).Value = endD        ' EndDate
    led.Cells(r, 3).Value = bucket      ' Bucket
    led.Cells(r, 4).Value = tenor       ' Tenor(days)
    led.Cells(r, 5).Value = rateDate    ' RateDate
    ' col F (Rate%) intentionally blank
    led.Cells(r, 7).Value = notional    ' Notional($mm)
    ' col H (Interest) intentionally blank
End Sub

Public Sub BuildDepositLedger()
    Dim inp As Worksheet, led As Worksheet
    On Error GoTo Failed

    On Error Resume Next: Set inp = ThisWorkbook.Worksheets("INPUTS"): On Error GoTo Failed
    If inp Is Nothing Then MsgBox "INPUTS sheet not found.": Exit Sub
    LoadHolidays inp

    On Error Resume Next: Set led = ThisWorkbook.Worksheets("LEDGER"): On Error GoTo Failed
    If led Is Nothing Then
        Set led = ThisWorkbook.Worksheets.Add(After:=inp)
        led.Name = "LEDGER"
    Else
        led.Cells.Clear
    End If

    ' read inputs
    Dim startDate As Date, horizonYrs As Double
    Dim onAmt As Double, m1Amt As Double, m2Amt As Double, m3Amt As Double
    startDate    = CDate(inp.Range("B4").Value)
    horizonYrs   = CDbl(inp.Range("B5").Value)
    onAmt        = CDbl(inp.Range("B6").Value)
    m1Amt        = CDbl(inp.Range("B7").Value)
    m2Amt        = CDbl(inp.Range("B8").Value)
    m3Amt        = CDbl(inp.Range("B9").Value)
    Dim horizonMonths As Long: horizonMonths = CLng(horizonYrs * 12)
    Dim endDate As Date:       endDate = AddMonths(startDate, horizonMonths)
    startDate = Following(startDate)

    ' header
    Dim hdr As Variant
    hdr = Array("StartDate","EndDate","Bucket","Tenor(days)","RateDate", _
                "Rate%","Notional($mm)","Interest($mm)")
    Dim c As Long
    For c = 0 To UBound(hdr)
        With led.Cells(1, c + 1): .Value = hdr(c): .Font.Bold = True: End With
    Next c

    Dim row As Long: row = 2

    ' term bucket state: rate lock date and next maturity
    Dim m1RD As Date, m1Mat As Date
    Dim m2RD As Date, m2Mat As Date
    Dim m3RD As Date, m3Mat As Date
    If m1Amt > 0 Then m1RD = startDate: m1Mat = Following(AddMonths(startDate, 1))
    If m2Amt > 0 Then m2RD = startDate: m2Mat = Following(AddMonths(startDate, 2))
    If m3Amt > 0 Then m3RD = startDate: m3Mat = Following(AddMonths(startDate, 3))

    ' walk every business day
    Dim d As Date: d = startDate
    Do While d < endDate
        Dim nxt As Date: nxt = NextBD(d)
        If nxt > endDate Then nxt = endDate
        Dim onTenor As Long: onTenor = CLng(nxt - d)

        ' ON: StartDate=d, EndDate=nextBD, tenor=1/3/4, RateDate=d
        If onAmt > 0 Then
            WriteRow led, row, d, nxt, "ON", onTenor, d, onAmt: row = row + 1
        End If

        ' 1M: roll at maturity, RateDate locks at placement
        If m1Amt > 0 Then
            If d >= m1Mat Then m1RD = d: m1Mat = Following(AddMonths(d, 1))
            WriteRow led, row, d, nxt, "1M", onTenor, m1RD, m1Amt: row = row + 1
        End If

        ' 2M
        If m2Amt > 0 Then
            If d >= m2Mat Then m2RD = d: m2Mat = Following(AddMonths(d, 2))
            WriteRow led, row, d, nxt, "2M", onTenor, m2RD, m2Amt: row = row + 1
        End If

        ' 3M
        If m3Amt > 0 Then
            If d >= m3Mat Then m3RD = d: m3Mat = Following(AddMonths(d, 3))
            WriteRow led, row, d, nxt, "3M", onTenor, m3RD, m3Amt: row = row + 1
        End If

        d = nxt
    Loop

    ' format
    led.Columns("A:B").NumberFormat = "yyyy-mm-dd"
    led.Columns("E").NumberFormat   = "yyyy-mm-dd"
    led.Columns("G").NumberFormat   = "#,##0.0"
    led.Columns("F:F").NumberFormat = "0.0000"
    led.Columns("H:H").NumberFormat = "0.000000"
    led.Columns("A:H").AutoFit
    led.Activate

    MsgBox "Deposit ledger built: " & (row - 2) & " rows." & vbNewLine & _
           "Wire Rate%  (F2): =RCurveRate(curveCell, E2)" & vbNewLine & _
           "Wire Interest (H2): =G2*F2/100*D2/360"
    Exit Sub
Failed:
    MsgBox "BuildDepositLedger error: " & Err.Description
End Sub
