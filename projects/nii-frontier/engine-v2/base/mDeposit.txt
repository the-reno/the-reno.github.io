' =====================================================================
' mDeposit  -  deposit daily interest skeleton
'
' RUN: Alt+F8 > BuildDepositLedger
'
' One row per business day per bucket showing the interest accruing
' that day. No cash in/out events - pure daily income skeleton.
' Rate% and Interest($mm) are LEFT BLANK: wire RCurveRate and RAccrue
' to those columns after connecting a curve.
'
' MECHANICS
'   ON  : one row per business day. Tenor = calendar days to next BD
'         (1 normally, 3 on Fridays, more over holiday bridges).
'         RateDate = that day (rate fixes each morning).
'   1M/2M/3M: one row per business day within each deposit period.
'         Tenor = 1 (each row is one day's accrual).
'         RateDate = the placement date of that period (rate locked
'         at inception - a 1M deposit captures the rate at start).
'         Deposit period rolls to Following(PlaceDate + nMonths).
'
' INPUT CELL MAP (INPUTS sheet)
'   B4  Start date
'   B5  Horizon (years: 0.5 / 1 / 1.5 / 2)
'   B6  ON allocation  ($mm)
'   B7  1M allocation  ($mm)
'   B8  2M allocation  ($mm)
'   B9  3M allocation  ($mm)
'   A14:A30  Holiday dates (one per row, blank row stops reading)
'
' LEDGER COLUMNS
'   A  Date            B  Bucket         C  Notional($mm)
'   D  Tenor(days)     E  RateDate
'   F  Rate%           G  Interest($mm)
'   (F and G are blank - fill with =RCurveRate and =RAccrue)
' =====================================================================
Option Explicit

Private Hol As Object

Private Sub LoadHolidays(ws As Worksheet)
    Set Hol = CreateObject("Scripting.Dictionary")
    Dim r As Long, v As Variant
    For r = 14 To 50
        v = ws.Cells(r, 1).Value
        If v = "" Then Exit For
        If IsDate(v) Then
            Dim k As Long: k = CLng(CDate(v))
            If Not Hol.Exists(k) Then Hol.Add k, True
        End If
    Next r
End Sub

Private Function IsBusinessDay(d As Date) As Boolean
    Dim w As Long: w = Weekday(d, vbMonday)
    IsBusinessDay = (w <= 5) And Not Hol.Exists(CLng(d))
End Function

Private Function Following(d As Date) As Date
    Do While Not IsBusinessDay(d): d = d + 1: Loop
    Following = d
End Function

Private Function NextBD(d As Date) As Date
    NextBD = Following(d + 1)
End Function

Private Function AddMonths(d As Date, m As Long) As Date
    Dim y As Long, mo As Long, dy As Long
    y = Year(d): mo = Month(d): dy = Day(d)
    mo = mo + m
    Do While mo > 12: mo = mo - 12: y = y + 1: Loop
    Dim last As Long: last = Day(DateSerial(y, mo + 1, 0))
    If dy > last Then dy = last
    AddMonths = DateSerial(y, mo, dy)
End Function

' ---- write one daily accrual row
Private Sub WriteRow(led As Worksheet, r As Long, _
                     d As Date, bucket As String, notional As Double, _
                     tenor As Long, rateDate As Date)
    led.Cells(r, 1).Value = d           ' Date
    led.Cells(r, 2).Value = bucket      ' Bucket
    led.Cells(r, 3).Value = notional    ' Notional($mm)
    led.Cells(r, 4).Value = tenor       ' Tenor(days) - how long this rate applies
    led.Cells(r, 5).Value = rateDate    ' RateDate - rate locks here
    ' cols F (Rate%) and G (Interest) intentionally blank
End Sub

Public Sub BuildDepositLedger()
    Dim inp As Worksheet, led As Worksheet
    On Error GoTo Failed

    On Error Resume Next: Set inp = ThisWorkbook.Worksheets("INPUTS"): On Error GoTo Failed
    If inp Is Nothing Then MsgBox "INPUTS sheet not found.": Exit Sub
    LoadHolidays inp

    On Error Resume Next: Set led = ThisWorkbook.Worksheets("LEDGER"): On Error GoTo Failed
    If led Is Nothing Then
        Set led = ThisWorkbook.Worksheets.Add(After:=inp)
        led.Name = "LEDGER"
    Else
        led.Cells.Clear
    End If

    ' --- read inputs
    Dim startDate As Date, horizonYrs As Double
    Dim onAmt As Double, m1Amt As Double, m2Amt As Double, m3Amt As Double
    startDate   = CDate(inp.Range("B4").Value)
    horizonYrs  = CDbl(inp.Range("B5").Value)
    onAmt       = CDbl(inp.Range("B6").Value)
    m1Amt       = CDbl(inp.Range("B7").Value)
    m2Amt       = CDbl(inp.Range("B8").Value)
    m3Amt       = CDbl(inp.Range("B9").Value)
    Dim horizonMonths As Long: horizonMonths = CLng(horizonYrs * 12)
    Dim endDate As Date: endDate = AddMonths(startDate, horizonMonths)
    startDate = Following(startDate)

    ' --- header
    Dim hdr As Variant
    hdr = Array("Date","Bucket","Notional($mm)","Tenor(days)","RateDate","Rate%","Interest($mm)")
    Dim c As Long
    For c = 0 To UBound(hdr)
        With led.Cells(1, c + 1): .Value = hdr(c): .Font.Bold = True: End With
    Next c

    Dim row As Long: row = 2

    ' =====================================================================
    ' Walk every business day in the horizon.
    ' For each day, write one row per active bucket.
    ' =====================================================================
    Dim d As Date: d = startDate

    ' term-bucket state: track the current period's rate-lock date
    Dim m1RateDate As Date, m1MatDate As Date
    Dim m2RateDate As Date, m2MatDate As Date
    Dim m3RateDate As Date, m3MatDate As Date

    ' initialise first period for each term bucket
    If m1Amt > 0 Then m1RateDate = d: m1MatDate = Following(AddMonths(d, 1))
    If m2Amt > 0 Then m2RateDate = d: m2MatDate = Following(AddMonths(d, 2))
    If m3Amt > 0 Then m3RateDate = d: m3MatDate = Following(AddMonths(d, 3))

    Do While d < endDate
        Dim nxt As Date: nxt = NextBD(d)
        If nxt > endDate Then nxt = endDate
        Dim onTenor As Long: onTenor = CLng(nxt - d)

        ' ON: tenor = days to next BD (captures bridges naturally)
        If onAmt > 0 Then
            WriteRow led, row, d, "ON", onAmt, onTenor, d: row = row + 1
        End If

        ' 1M: roll if we've reached or passed the maturity date
        If m1Amt > 0 Then
            If d >= m1MatDate And d < endDate Then
                m1RateDate = d: m1MatDate = Following(AddMonths(d, 1))
            End If
            WriteRow led, row, d, "1M", m1Amt, 1, m1RateDate: row = row + 1
        End If

        ' 2M
        If m2Amt > 0 Then
            If d >= m2MatDate And d < endDate Then
                m2RateDate = d: m2MatDate = Following(AddMonths(d, 2))
            End If
            WriteRow led, row, d, "2M", m2Amt, 1, m2RateDate: row = row + 1
        End If

        ' 3M
        If m3Amt > 0 Then
            If d >= m3MatDate And d < endDate Then
                m3RateDate = d: m3MatDate = Following(AddMonths(d, 3))
            End If
            WriteRow led, row, d, "3M", m3Amt, 1, m3RateDate: row = row + 1
        End If

        d = nxt
    Loop

    ' --- format
    led.Columns("A").NumberFormat = "yyyy-mm-dd"
    led.Columns("E").NumberFormat = "yyyy-mm-dd"
    led.Columns("C").NumberFormat = "#,##0.0"
    led.Columns("F:G").NumberFormat = "0.000000"
    led.Columns("A:G").AutoFit
    led.Activate

    MsgBox "Deposit ledger built: " & (row - 2) & " rows written to LEDGER." & vbNewLine & _
           "Wire Rate% (col F) with =RCurveRate(curve,E2)" & vbNewLine & _
           "and Interest (col G) with =RAccrue(A2,A2+D2,C2,""SIMPLE"",curve)"
    Exit Sub
Failed:
    MsgBox "BuildDepositLedger error: " & Err.Description
End Sub
