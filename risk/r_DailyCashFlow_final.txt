Option Explicit

'====================================================================
' r_DailyCashFlow
'
' Excel formula:
' =r_DailyCashFlow(AnalysisDate, LookbackYears, InitialCash, CurveRange)
'
' Curve layout:
' Column 1 = Date
' Column 2 = ON
' Column 3 = 1M
' Column 4 = 2M
' Column 5 = 3M
' Column 6 = 6M
'
' Roll convention:
' 1. AnalysisDate is the exact common final maturity date.
' 2. The final maturity must exist in the curve.
' 3. Each prior roll target is calculated backward from the following
'    maturity using DateAdd("m", -tenor, maturity).
' 4. The actual roll date is the latest curve date on or before that target.
' 5. The requested lookback date is only a boundary.
' 6. The first retained roll date is the first scheduled roll date on or
'    after the requested lookback boundary.
' 7. No artificial stub roll is inserted at the lookback boundary.
' 8. Roll = TRUE only on actual investment or reinvestment dates.
' 9. AnalysisDate is a maturity date, not a new roll date.
'====================================================================

Public Function r_DailyCashFlow( _
    ByVal AnalysisDate As Variant, _
    ByVal LookbackYears As Double, _
    ByVal InitialCash As Double, _
    ByVal CurveRange As Range) As Variant

    Dim DataRange As Range
    Dim Curve As Variant
    Dim Result() As Variant

    Dim RollDates(2 To 5) As Variant
    Dim TenorName(1 To 5) As String
    Dim TenorMonths(1 To 5) As Long
    Dim RateColumn(1 To 5) As Long

    Dim Principal(1 To 5) As Double
    Dim Accrued(1 To 5) As Double
    Dim LockedRate(1 To 5) As Double
    Dim Maturity(1 To 5) As Date
    Dim IsActive(1 To 5) As Boolean

    Dim ExactAnalysisEnd As Date
    Dim RequestedStart As Date
    Dim OutputStart As Date
    Dim CurrentDate As Date
    Dim CurrentCurveDate As Date

    Dim FirstCurveRow As Long
    Dim LastCurveRow As Long
    Dim CurrentCurveRow As Long
    Dim FirstDataRow As Long
    Dim LastDataRow As Long

    Dim NumberOfDays As Long
    Dim DayNumber As Long
    Dim OutputRow As Long
    Dim OutputColumn As Long
    Dim Tenor As Long

    Dim DailyAccrual As Double
    Dim PortfolioValue As Double
    Dim ONValue As Double
    Dim RollFlag As Boolean

    On Error GoTo FunctionError

    If Not IsDate(AnalysisDate) And Not IsNumeric(AnalysisDate) Then
        r_DailyCashFlow = "Analysis date is not a valid Excel date."
        Exit Function
    End If

    If LookbackYears <= 0 Then
        r_DailyCashFlow = "Lookback years must be greater than zero."
        Exit Function
    End If

    If InitialCash = 0 Then
        r_DailyCashFlow = "Initial cash cannot be zero."
        Exit Function
    End If

    If CurveRange.Columns.Count < 6 Then
        r_DailyCashFlow = "Curve range must include Date, ON, 1M, 2M, 3M and 6M."
        Exit Function
    End If

    FirstDataRow = CurveRange.Row

    LastDataRow = CurveRange.Worksheet.Cells( _
        CurveRange.Worksheet.Rows.Count, _
        CurveRange.Column).End(xlUp).Row

    If LastDataRow < FirstDataRow Then
        r_DailyCashFlow = "No curve data was found."
        Exit Function
    End If

    Set DataRange = CurveRange.Worksheet.Range( _
        CurveRange.Worksheet.Cells(FirstDataRow, CurveRange.Column), _
        CurveRange.Worksheet.Cells(LastDataRow, CurveRange.Column + 5))

    Curve = DataRange.Value2
    LastCurveRow = UBound(Curve, 1)

    If IsValidCurveDate(Curve(1, 1)) Then
        FirstCurveRow = 1
    Else
        FirstCurveRow = 2
    End If

    If LastCurveRow < FirstCurveRow Then
        r_DailyCashFlow = "No valid curve rows were found."
        Exit Function
    End If

    If Not ValidateCurveDates(Curve, FirstCurveRow, LastCurveRow) Then
        r_DailyCashFlow = "Curve dates must be valid and sorted from oldest to newest."
        Exit Function
    End If

    RateColumn(1) = 2
    RateColumn(2) = 3
    RateColumn(3) = 4
    RateColumn(4) = 5
    RateColumn(5) = 6

    TenorName(1) = "ON"
    TenorName(2) = "1M"
    TenorName(3) = "2M"
    TenorName(4) = "3M"
    TenorName(5) = "6M"

    TenorMonths(1) = 0
    TenorMonths(2) = 1
    TenorMonths(3) = 2
    TenorMonths(4) = 3
    TenorMonths(5) = 6

    ExactAnalysisEnd = ConvertToDate(AnalysisDate)

    If Not CurveDateExists( _
        ExactAnalysisEnd, Curve, FirstCurveRow, LastCurveRow) Then

        r_DailyCashFlow = "The analysis date must exist exactly in the curve."
        Exit Function
    End If

    RequestedStart = DateAdd( _
        "m", -CLng(LookbackYears * 12#), ExactAnalysisEnd)

    For Tenor = 2 To 5
        RollDates(Tenor) = BuildRollSchedule( _
            RequestedStart, ExactAnalysisEnd, TenorMonths(Tenor), _
            Curve, FirstCurveRow, LastCurveRow)

        If IsEmpty(RollDates(Tenor)) Then
            r_DailyCashFlow = "No retained roll dates were found for " & TenorName(Tenor) & "."
            Exit Function
        End If
    Next Tenor

    'Use the most recent first retained roll date as the common table start.
    OutputStart = FirstRollDate(RollDates(2))

    For Tenor = 3 To 5
        If FirstRollDate(RollDates(Tenor)) > OutputStart Then
            OutputStart = FirstRollDate(RollDates(Tenor))
        End If
    Next Tenor

    If OutputStart >= ExactAnalysisEnd Then
        r_DailyCashFlow = "The retained roll schedule does not contain a valid analysis period."
        Exit Function
    End If

    NumberOfDays = DateDiff("d", OutputStart, ExactAnalysisEnd) + 1

    ReDim Result(1 To NumberOfDays + 1, 1 To 26)

    Result(1, 1) = "Date"
    OutputColumn = 2

    For Tenor = 1 To 5
        Result(1, OutputColumn) = TenorName(Tenor) & " Rate"
        Result(1, OutputColumn + 1) = TenorName(Tenor) & " Maturity"
        Result(1, OutputColumn + 2) = TenorName(Tenor) & " Daily Accrual"
        Result(1, OutputColumn + 3) = TenorName(Tenor) & " Value"
        Result(1, OutputColumn + 4) = TenorName(Tenor) & " Roll"
        OutputColumn = OutputColumn + 5
    Next Tenor

    CurrentCurveRow = FindCurveRow( _
        OutputStart, Curve, FirstCurveRow, LastCurveRow)

    For DayNumber = 0 To NumberOfDays - 1

        CurrentDate = OutputStart + DayNumber
        OutputRow = DayNumber + 2
        Result(OutputRow, 1) = CurrentDate

        Do While CurrentCurveRow < LastCurveRow
            If CurveDateValue(Curve(CurrentCurveRow + 1, 1)) <= CurrentDate Then
                CurrentCurveRow = CurrentCurveRow + 1
            Else
                Exit Do
            End If
        Loop

        CurrentCurveDate = CurveDateValue(Curve(CurrentCurveRow, 1))

        '-----------------------------
        ' ON strategy
        '-----------------------------
        LockedRate(1) = GetRateValue(Curve(CurrentCurveRow, RateColumn(1)))

        If CurrentDate = OutputStart Then
            Principal(1) = InitialCash
            ONValue = InitialCash
        Else
            Principal(1) = ONValue
        End If

        RollFlag = (CurrentDate = CurrentCurveDate And CurrentDate < ExactAnalysisEnd)

        If CurrentDate < ExactAnalysisEnd Then
            DailyAccrual = Principal(1) * LockedRate(1) / 100# / 360#
            ONValue = Principal(1) + DailyAccrual
        Else
            DailyAccrual = 0
            ONValue = Principal(1)
            RollFlag = False
        End If

        Result(OutputRow, 2) = LockedRate(1)
        Result(OutputRow, 3) = NextCurveDate( _
            CurrentDate, ExactAnalysisEnd, Curve, FirstCurveRow, LastCurveRow)
        Result(OutputRow, 4) = DailyAccrual
        Result(OutputRow, 5) = ONValue
        Result(OutputRow, 6) = RollFlag

        '-----------------------------
        ' Fixed-tenor strategies
        '-----------------------------
        OutputColumn = 7

        For Tenor = 2 To 5

            RollFlag = IsScheduledRollDate(CurrentDate, RollDates(Tenor))

            If Not IsActive(Tenor) Then

                If RollFlag Then
                    IsActive(Tenor) = True
                    Principal(Tenor) = InitialCash
                    Accrued(Tenor) = 0
                    LockedRate(Tenor) = GetRateValue( _
                        Curve(CurrentCurveRow, RateColumn(Tenor)))
                    Maturity(Tenor) = NextMaturityDate( _
                        CurrentDate, RollDates(Tenor), ExactAnalysisEnd)
                End If

            ElseIf RollFlag Then

                Principal(Tenor) = Principal(Tenor) + Accrued(Tenor)
                Accrued(Tenor) = 0
                LockedRate(Tenor) = GetRateValue( _
                    Curve(CurrentCurveRow, RateColumn(Tenor)))
                Maturity(Tenor) = NextMaturityDate( _
                    CurrentDate, RollDates(Tenor), ExactAnalysisEnd)

            ElseIf CurrentDate = ExactAnalysisEnd Then

                Principal(Tenor) = Principal(Tenor) + Accrued(Tenor)
                Accrued(Tenor) = 0
                Maturity(Tenor) = ExactAnalysisEnd
                RollFlag = False

            End If

            If IsActive(Tenor) Then

                If CurrentDate < ExactAnalysisEnd Then
                    DailyAccrual = Principal(Tenor) * LockedRate(Tenor) / 100# / 360#
                    Accrued(Tenor) = Accrued(Tenor) + DailyAccrual
                Else
                    DailyAccrual = 0
                End If

                PortfolioValue = Principal(Tenor) + Accrued(Tenor)

                Result(OutputRow, OutputColumn) = LockedRate(Tenor)
                Result(OutputRow, OutputColumn + 1) = Maturity(Tenor)
                Result(OutputRow, OutputColumn + 2) = DailyAccrual
                Result(OutputRow, OutputColumn + 3) = PortfolioValue
                Result(OutputRow, OutputColumn + 4) = RollFlag

            Else

                Result(OutputRow, OutputColumn) = Empty
                Result(OutputRow, OutputColumn + 1) = Empty
                Result(OutputRow, OutputColumn + 2) = Empty
                Result(OutputRow, OutputColumn + 3) = Empty
                Result(OutputRow, OutputColumn + 4) = False

            End If

            OutputColumn = OutputColumn + 5

        Next Tenor

    Next DayNumber

    r_DailyCashFlow = Result
    Exit Function

FunctionError:
    r_DailyCashFlow = "r_DailyCashFlow error: " & Err.Number & " - " & Err.Description

End Function

Private Function BuildRollSchedule( _
    ByVal RequestedStart As Date, _
    ByVal AnalysisEnd As Date, _
    ByVal MonthsInTenor As Long, _
    ByVal Curve As Variant, _
    ByVal FirstCurveRow As Long, _
    ByVal LastCurveRow As Long) As Variant

    Const MAX_ROLLS As Long = 2000

    Dim BackwardDates(1 To MAX_ROLLS) As Double
    Dim FinalDates() As Double

    Dim CurrentMaturity As Date
    Dim TargetRollDate As Date
    Dim ActualRollDate As Date

    Dim Count As Long
    Dim KeepCount As Long
    Dim Position As Long
    Dim SourcePosition As Long

    CurrentMaturity = AnalysisEnd
    Count = 0

    Do
        TargetRollDate = DateAdd("m", -MonthsInTenor, CurrentMaturity)

        ActualRollDate = PreviousCurveDate( _
            TargetRollDate, Curve, FirstCurveRow, LastCurveRow)

        If ActualRollDate = 0 Then Exit Do
        If ActualRollDate >= CurrentMaturity Then Exit Do

        Count = Count + 1
        BackwardDates(Count) = CDbl(ActualRollDate)
        CurrentMaturity = ActualRollDate

        If ActualRollDate < RequestedStart Then Exit Do
        If Count >= MAX_ROLLS Then Exit Do
    Loop

    KeepCount = 0

    For Position = Count To 1 Step -1
        If CDate(BackwardDates(Position)) >= RequestedStart Then
            KeepCount = KeepCount + 1
        End If
    Next Position

    If KeepCount = 0 Then
        BuildRollSchedule = Empty
        Exit Function
    End If

    ReDim FinalDates(1 To KeepCount)
    Position = 0

    For SourcePosition = Count To 1 Step -1
        If CDate(BackwardDates(SourcePosition)) >= RequestedStart Then
            Position = Position + 1
            FinalDates(Position) = BackwardDates(SourcePosition)
        End If
    Next SourcePosition

    BuildRollSchedule = FinalDates

End Function

Private Function FirstRollDate(ByVal RollArray As Variant) As Date
    FirstRollDate = CDate(RollArray(LBound(RollArray)))
End Function

Private Function IsScheduledRollDate( _
    ByVal TargetDate As Date, _
    ByVal RollArray As Variant) As Boolean

    Dim Position As Long

    IsScheduledRollDate = False

    If IsEmpty(RollArray) Then Exit Function

    For Position = LBound(RollArray) To UBound(RollArray)
        If CLng(CDbl(RollArray(Position))) = CLng(CDbl(TargetDate)) Then
            IsScheduledRollDate = True
            Exit Function
        End If
    Next Position

End Function

Private Function NextMaturityDate( _
    ByVal CurrentDate As Date, _
    ByVal RollArray As Variant, _
    ByVal AnalysisEnd As Date) As Date

    Dim Position As Long

    For Position = LBound(RollArray) To UBound(RollArray)
        If CLng(CDbl(RollArray(Position))) > CLng(CDbl(CurrentDate)) Then
            NextMaturityDate = CDate(RollArray(Position))
            Exit Function
        End If
    Next Position

    NextMaturityDate = AnalysisEnd

End Function

Private Function CurveDateExists( _
    ByVal TargetDate As Date, _
    ByVal Curve As Variant, _
    ByVal FirstCurveRow As Long, _
    ByVal LastCurveRow As Long) As Boolean

    Dim RowNumber As Long
    Dim CandidateDate As Date

    CurveDateExists = False

    For RowNumber = FirstCurveRow To LastCurveRow
        CandidateDate = CurveDateValue(Curve(RowNumber, 1))

        If CLng(CandidateDate) = CLng(TargetDate) Then
            CurveDateExists = True
            Exit Function
        End If

        If CandidateDate > TargetDate Then Exit For
    Next RowNumber

End Function

Private Function PreviousCurveDate( _
    ByVal TargetDate As Date, _
    ByVal Curve As Variant, _
    ByVal FirstCurveRow As Long, _
    ByVal LastCurveRow As Long) As Date

    Dim RowNumber As Long
    Dim CandidateDate As Date

    PreviousCurveDate = 0

    For RowNumber = LastCurveRow To FirstCurveRow Step -1
        CandidateDate = CurveDateValue(Curve(RowNumber, 1))

        If CandidateDate <= TargetDate Then
            PreviousCurveDate = CandidateDate
            Exit Function
        End If
    Next RowNumber

End Function

Private Function NextCurveDate( _
    ByVal CurrentDate As Date, _
    ByVal AnalysisEnd As Date, _
    ByVal Curve As Variant, _
    ByVal FirstCurveRow As Long, _
    ByVal LastCurveRow As Long) As Date

    Dim RowNumber As Long
    Dim CandidateDate As Date

    If CurrentDate >= AnalysisEnd Then
        NextCurveDate = AnalysisEnd
        Exit Function
    End If

    For RowNumber = FirstCurveRow To LastCurveRow
        CandidateDate = CurveDateValue(Curve(RowNumber, 1))

        If CandidateDate > CurrentDate Then
            If CandidateDate > AnalysisEnd Then
                NextCurveDate = AnalysisEnd
            Else
                NextCurveDate = CandidateDate
            End If
            Exit Function
        End If
    Next RowNumber

    NextCurveDate = AnalysisEnd

End Function

Private Function FindCurveRow( _
    ByVal TargetDate As Date, _
    ByVal Curve As Variant, _
    ByVal FirstCurveRow As Long, _
    ByVal LastCurveRow As Long) As Long

    Dim RowNumber As Long
    Dim CandidateDate As Date

    FindCurveRow = FirstCurveRow

    For RowNumber = FirstCurveRow To LastCurveRow
        CandidateDate = CurveDateValue(Curve(RowNumber, 1))

        If CandidateDate <= TargetDate Then
            FindCurveRow = RowNumber
        Else
            Exit For
        End If
    Next RowNumber

End Function

Private Function ValidateCurveDates( _
    ByVal Curve As Variant, _
    ByVal FirstCurveRow As Long, _
    ByVal LastCurveRow As Long) As Boolean

    Dim RowNumber As Long
    Dim PreviousDate As Date
    Dim CurrentDate As Date

    ValidateCurveDates = False

    If Not IsValidCurveDate(Curve(FirstCurveRow, 1)) Then Exit Function

    PreviousDate = CurveDateValue(Curve(FirstCurveRow, 1))

    For RowNumber = FirstCurveRow + 1 To LastCurveRow
        If Not IsValidCurveDate(Curve(RowNumber, 1)) Then Exit Function

        CurrentDate = CurveDateValue(Curve(RowNumber, 1))

        If CurrentDate <= PreviousDate Then Exit Function

        PreviousDate = CurrentDate
    Next RowNumber

    ValidateCurveDates = True

End Function

Private Function IsValidCurveDate(ByVal ValueToCheck As Variant) As Boolean

    IsValidCurveDate = False

    If IsError(ValueToCheck) Then Exit Function
    If IsEmpty(ValueToCheck) Then Exit Function

    If IsNumeric(ValueToCheck) Then
        If CDbl(ValueToCheck) > 0 Then IsValidCurveDate = True
        Exit Function
    End If

    If Len(Trim$(CStr(ValueToCheck))) = 0 Then Exit Function

    If IsDate(ValueToCheck) Then IsValidCurveDate = True

End Function

Private Function CurveDateValue(ByVal ValueToConvert As Variant) As Date

    If IsNumeric(ValueToConvert) Then
        CurveDateValue = DateSerial(1899, 12, 30) + CLng(CDbl(ValueToConvert))
    Else
        CurveDateValue = CDate(ValueToConvert)
    End If

End Function

Private Function ConvertToDate(ByVal ValueToConvert As Variant) As Date

    If IsNumeric(ValueToConvert) Then
        ConvertToDate = DateSerial(1899, 12, 30) + CLng(CDbl(ValueToConvert))
    Else
        ConvertToDate = CDate(ValueToConvert)
    End If

End Function

Private Function GetRateValue(ByVal RateValue As Variant) As Double

    If IsError(RateValue) Then
        Err.Raise vbObjectError + 1001, "r_DailyCashFlow", _
            "A curve rate contains an Excel error."
    ElseIf IsEmpty(RateValue) Then
        Err.Raise vbObjectError + 1002, "r_DailyCashFlow", _
            "A required curve rate is blank."
    ElseIf Not IsNumeric(RateValue) Then
        Err.Raise vbObjectError + 1003, "r_DailyCashFlow", _
            "A curve rate is not numeric."
    Else
        GetRateValue = CDbl(RateValue)
    End If

End Function

Public Sub Format_r_DailyCashFlow()

    Dim OutputRange As Range
    Dim FirstColumn As Long

    On Error GoTo FormatError

    Set OutputRange = Selection.CurrentRegion

    OutputRange.Columns(1).NumberFormat = "dd-mm-yy"

    For FirstColumn = 2 To 26 Step 5
        OutputRange.Columns(FirstColumn).NumberFormat = "0.0000"
        OutputRange.Columns(FirstColumn + 1).NumberFormat = "dd-mm-yy"
        OutputRange.Columns(FirstColumn + 2).NumberFormat = "#,##0.0000"
        OutputRange.Columns(FirstColumn + 3).NumberFormat = "#,##0.0000"
        OutputRange.Columns(FirstColumn + 4).NumberFormat = "General"
    Next FirstColumn

    OutputRange.Columns.AutoFit
    Exit Sub

FormatError:
    MsgBox "Formatting error: " & Err.Description, vbExclamation, _
        "r_DailyCashFlow"

End Sub
