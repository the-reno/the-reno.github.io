Option Explicit

Private Const SOURCE_SHEET As String = "Sheet15"
Private Const OUTPUT_SHEET As String = "Tenor Analysis"

Private Const RAW_DATE_COL As Long = 1       'A
Private Const RAW_ON_COL As Long = 2         'B
Private Const FIRST_RAW_ROW As Long = 2

Private Const ROLL_DATE_COL As Long = 9      'I
Private Const FIRST_ROLL_ROW As Long = 3


Public Sub CreateAllTenorAnalysis()

    Dim wsSource As Worksheet
    Dim wsOutput As Worksheet
    Dim lastRawRow As Long
    Dim lastRollRow As Long

    Application.ScreenUpdating = False
    Application.EnableEvents = False

    On Error GoTo ErrorHandler

    Set wsSource = ThisWorkbook.Worksheets(SOURCE_SHEET)
    Set wsOutput = GetOrCreateOutputSheet(OUTPUT_SHEET)

    lastRawRow = wsSource.Cells(wsSource.Rows.Count, RAW_DATE_COL).End(xlUp).Row
    lastRollRow = wsSource.Cells(wsSource.Rows.Count, ROLL_DATE_COL).End(xlUp).Row

    ClearPreviousAnalysis wsOutput

    CreateAnalysisHeader wsOutput

    BuildTenorSection _
        wsSource:=wsSource, _
        wsOutput:=wsOutput, _
        tenorName:="1M", _
        tenorMonths:=1, _
        rawTenorColumn:=3, _
        outputStartColumn:=1, _
        chartLeftColumn:=7, _
        chartTopRow:=3, _
        lastRawRow:=lastRawRow, _
        lastRollRow:=lastRollRow

    BuildTenorSection _
        wsSource:=wsSource, _
        wsOutput:=wsOutput, _
        tenorName:="2M", _
        tenorMonths:=2, _
        rawTenorColumn:=4, _
        outputStartColumn:=1, _
        chartLeftColumn:=7, _
        chartTopRow:=29, _
        lastRawRow:=lastRawRow, _
        lastRollRow:=lastRollRow

    BuildTenorSection _
        wsSource:=wsSource, _
        wsOutput:=wsOutput, _
        tenorName:="3M", _
        tenorMonths:=3, _
        rawTenorColumn:=5, _
        outputStartColumn:=17, _
        chartLeftColumn:=23, _
        chartTopRow:=3, _
        lastRawRow:=lastRawRow, _
        lastRollRow:=lastRollRow

    BuildTenorSection _
        wsSource:=wsSource, _
        wsOutput:=wsOutput, _
        tenorName:="6M", _
        tenorMonths:=6, _
        rawTenorColumn:=6, _
        outputStartColumn:=17, _
        chartLeftColumn:=23, _
        chartTopRow:=29, _
        lastRawRow:=lastRawRow, _
        lastRollRow:=lastRollRow

    FormatOutputSheet wsOutput

    wsOutput.Activate
    wsOutput.Range("A1").Select

    Application.ScreenUpdating = True
    Application.EnableEvents = True

    MsgBox "The 1M, 2M, 3M and 6M analysis was created.", vbInformation

    Exit Sub

ErrorHandler:

    Application.ScreenUpdating = True
    Application.EnableEvents = True

    MsgBox _
        "The analysis could not be completed." & vbCrLf & vbCrLf & _
        "Error " & Err.Number & ": " & Err.Description, _
        vbCritical

End Sub


Private Sub BuildTenorSection( _
    ByVal wsSource As Worksheet, _
    ByVal wsOutput As Worksheet, _
    ByVal tenorName As String, _
    ByVal tenorMonths As Long, _
    ByVal rawTenorColumn As Long, _
    ByVal outputStartColumn As Long, _
    ByVal chartLeftColumn As Long, _
    ByVal chartTopRow As Long, _
    ByVal lastRawRow As Long, _
    ByVal lastRollRow As Long)

    Dim titleRow As Long
    Dim headerRow As Long
    Dim firstDataRow As Long
    Dim outputRow As Long
    Dim sourceRow As Long

    Dim rollDate As Date
    Dim maturityDate As Date

    Dim onAtStart As Variant
    Dim tenorRate As Variant
    Dim realizedON As Variant

    Dim maturityDays As Long
    Dim premiumBps As Double
    Dim pricingErrorBps As Double

    titleRow = chartTopRow
    headerRow = chartTopRow + 1
    firstDataRow = chartTopRow + 2
    outputRow = firstDataRow

    CreateSectionHeaders _
        ws:=wsOutput, _
        startColumn:=outputStartColumn, _
        titleRow:=titleRow, _
        headerRow:=headerRow, _
        tenorName:=tenorName

    For sourceRow = FIRST_ROLL_ROW To lastRollRow

        If IsDate(wsSource.Cells(sourceRow, ROLL_DATE_COL).Value) Then

            rollDate = CDate(wsSource.Cells(sourceRow, ROLL_DATE_COL).Value)
            maturityDate = DateAdd("m", tenorMonths, rollDate)

            If DataAvailableThroughMaturity( _
                wsSource, maturityDate, lastRawRow) Then

                onAtStart = GetRateForDate( _
                    ws:=wsSource, _
                    targetDate:=rollDate, _
                    rateColumn:=RAW_ON_COL, _
                    lastRawRow:=lastRawRow)

                tenorRate = GetRateForDate( _
                    ws:=wsSource, _
                    targetDate:=rollDate, _
                    rateColumn:=rawTenorColumn, _
                    lastRawRow:=lastRawRow)

                realizedON = CompoundedONRate( _
                    ws:=wsSource, _
                    startDate:=rollDate, _
                    maturityDate:=maturityDate, _
                    lastRawRow:=lastRawRow)

                If Not IsError(onAtStart) And _
                   Not IsError(tenorRate) And _
                   Not IsError(realizedON) Then

                    maturityDays = DateDiff("d", rollDate, maturityDate)

                    premiumBps = _
                        (CDbl(tenorRate) - CDbl(onAtStart)) * 100#

                    pricingErrorBps = _
                        (CDbl(tenorRate) - CDbl(realizedON)) * 100#

                    With wsOutput

                        .Cells(outputRow, outputStartColumn).Value = rollDate
                        .Cells(outputRow, outputStartColumn + 1).Value = maturityDate

                        'Convert percentage-point values into Excel percentages.
                        .Cells(outputRow, outputStartColumn + 2).Value = _
                            CDbl(onAtStart) / 100#

                        .Cells(outputRow, outputStartColumn + 3).Value = _
                            CDbl(tenorRate) / 100#

                        .Cells(outputRow, outputStartColumn + 4).Value = _
                            CDbl(realizedON) / 100#

                        .Cells(outputRow, outputStartColumn + 5).Value = premiumBps
                        .Cells(outputRow, outputStartColumn + 6).Value = _
                            pricingErrorBps

                        If pricingErrorBps > 0 Then
                            .Cells(outputRow, outputStartColumn + 7).Value = _
                                tenorName & " Outperformed"
                        ElseIf pricingErrorBps < 0 Then
                            .Cells(outputRow, outputStartColumn + 7).Value = _
                                "ON Outperformed"
                        Else
                            .Cells(outputRow, outputStartColumn + 7).Value = _
                                "Equal"
                        End If

                        .Cells(outputRow, outputStartColumn + 8).Value = _
                            maturityDays

                    End With

                    outputRow = outputRow + 1

                End If

            End If

        End If

    Next sourceRow

    If outputRow > firstDataRow Then

        FormatTenorTable _
            ws:=wsOutput, _
            startColumn:=outputStartColumn, _
            headerRow:=headerRow, _
            firstDataRow:=firstDataRow, _
            lastDataRow:=outputRow - 1

        CreateTenorChart _
            ws:=wsOutput, _
            tenorName:=tenorName, _
            startColumn:=outputStartColumn, _
            firstDataRow:=firstDataRow, _
            lastDataRow:=outputRow - 1, _
            chartLeftColumn:=chartLeftColumn, _
            chartTopRow:=chartTopRow

    Else

        wsOutput.Cells(firstDataRow, outputStartColumn).Value = _
            "No completed " & tenorName & " periods available."

    End If

End Sub


Private Sub CreateSectionHeaders( _
    ByVal ws As Worksheet, _
    ByVal startColumn As Long, _
    ByVal titleRow As Long, _
    ByVal headerRow As Long, _
    ByVal tenorName As String)

    With ws.Range( _
        ws.Cells(titleRow, startColumn), _
        ws.Cells(titleRow, startColumn + 8))

        .Merge
        .Value = tenorName & " Decision-Date Analysis"
        .Font.Bold = True
        .Font.Size = 13
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .Interior.Color = RGB(31, 78, 121)
        .Font.Color = RGB(255, 255, 255)

    End With

    ws.Cells(headerRow, startColumn).Value = "Roll Date"
    ws.Cells(headerRow, startColumn + 1).Value = "Maturity Date"
    ws.Cells(headerRow, startColumn + 2).Value = "ON at T0"
    ws.Cells(headerRow, startColumn + 3).Value = tenorName & " at T0"
    ws.Cells(headerRow, startColumn + 4).Value = "Realized ON"
    ws.Cells(headerRow, startColumn + 5).Value = "Premium (bp)"
    ws.Cells(headerRow, startColumn + 6).Value = "Pricing Error (bp)"
    ws.Cells(headerRow, startColumn + 7).Value = "Outcome"
    ws.Cells(headerRow, startColumn + 8).Value = "Days"

    With ws.Range( _
        ws.Cells(headerRow, startColumn), _
        ws.Cells(headerRow, startColumn + 8))

        .Font.Bold = True
        .Interior.Color = RGB(217, 225, 242)
        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter
        .Borders.LineStyle = xlContinuous
        .Borders.Color = RGB(180, 180, 180)

    End With

End Sub


Private Sub CreateTenorChart( _
    ByVal ws As Worksheet, _
    ByVal tenorName As String, _
    ByVal startColumn As Long, _
    ByVal firstDataRow As Long, _
    ByVal lastDataRow As Long, _
    ByVal chartLeftColumn As Long, _
    ByVal chartTopRow As Long)

    Dim chartObject As chartObject
    Dim chartName As String

    Dim dateRange As Range
    Dim onRange As Range
    Dim tenorRange As Range
    Dim realizedRange As Range

    chartName = "cht" & Replace(tenorName, " ", "")

    On Error Resume Next
    ws.ChartObjects(chartName).Delete
    On Error GoTo 0

    Set dateRange = ws.Range( _
        ws.Cells(firstDataRow, startColumn), _
        ws.Cells(lastDataRow, startColumn))

    Set onRange = ws.Range( _
        ws.Cells(firstDataRow, startColumn + 2), _
        ws.Cells(lastDataRow, startColumn + 2))

    Set tenorRange = ws.Range( _
        ws.Cells(firstDataRow, startColumn + 3), _
        ws.Cells(lastDataRow, startColumn + 3))

    Set realizedRange = ws.Range( _
        ws.Cells(firstDataRow, startColumn + 4), _
        ws.Cells(lastDataRow, startColumn + 4))

    Set chartObject = ws.ChartObjects.Add( _
        Left:=ws.Cells(chartTopRow, chartLeftColumn).Left, _
        Top:=ws.Cells(chartTopRow, chartLeftColumn).Top, _
        Width:=720, _
        Height:=360)

    chartObject.Name = chartName

    With chartObject.Chart

        .ChartType = xlLineMarkers

        Do While .SeriesCollection.Count > 0
            .SeriesCollection(1).Delete
        Loop

        'Series 1: ON at decision date.
        With .SeriesCollection.NewSeries

            .Name = "ON at Decision Date"
            .XValues = dateRange
            .Values = onRange
            .ChartType = xlLineMarkers

            .MarkerStyle = xlMarkerStyleCircle
            .MarkerSize = 5
            .MarkerForegroundColor = RGB(31, 78, 121)
            .MarkerBackgroundColor = RGB(255, 255, 255)

            .Format.Line.Visible = msoFalse

        End With

        'Series 2: tenor rate at decision date.
        With .SeriesCollection.NewSeries

            .Name = tenorName & " Rate at Decision Date"
            .XValues = dateRange
            .Values = tenorRange
            .ChartType = xlLineMarkers

            .MarkerStyle = xlMarkerStyleCircle
            .MarkerSize = 6
            .MarkerForegroundColor = RGB(31, 78, 121)
            .MarkerBackgroundColor = RGB(31, 78, 121)

            .Format.Line.Visible = msoFalse

        End With

        'Up/down premium bars.
        .ChartGroups(1).HasUpDownBars = True

        With .ChartGroups(1).UpBars.Format

            .Fill.Visible = msoTrue
            .Fill.Solid
            .Fill.ForeColor.RGB = RGB(91, 155, 213)
            .Fill.Transparency = 0.3

            .Line.Visible = msoTrue
            .Line.ForeColor.RGB = RGB(31, 78, 121)
            .Line.Transparency = 0.4

        End With

        'Negative premium: tenor below ON.
        With .ChartGroups(1).DownBars.Format

            .Fill.Visible = msoTrue
            .Fill.Solid
            .Fill.ForeColor.RGB = RGB(255, 99, 99)
            .Fill.Transparency = 0.25

            .Line.Visible = msoTrue
            .Line.ForeColor.RGB = RGB(192, 0, 0)
            .Line.Transparency = 0.35

        End With

        'Series 3: realized compounded ON.
        With .SeriesCollection.NewSeries

            .Name = "Realized Compounded ON"
            .XValues = dateRange
            .Values = realizedRange
            .ChartType = xlLineMarkers

            .MarkerStyle = xlMarkerStyleDiamond
            .MarkerSize = 6
            .MarkerForegroundColor = RGB(100, 100, 100)
            .MarkerBackgroundColor = RGB(100, 100, 100)

            .Format.Line.Visible = msoFalse

        End With

        .HasTitle = True
        .ChartTitle.Text = _
            tenorName & " Strategy – Decision-Date Perspective"

        .HasLegend = True
        .Legend.Position = xlLegendPositionBottom

        With .Axes(xlCategory)

            .HasTitle = True
            .AxisTitle.Text = "Roll Date"
            .TickLabels.NumberFormat = "mmm-yy"
            .TickLabelSpacing = 3

        End With

        With .Axes(xlValue)

            .HasTitle = True
            .AxisTitle.Text = "Annualized Rate"
            .TickLabels.NumberFormat = "0.00%"
            .HasMajorGridlines = True

            SetAutomaticYAxis _
                chartAxis:=.Parent.Axes(xlValue), _
                onRange:=onRange, _
                tenorRange:=tenorRange, _
                realizedRange:=realizedRange

        End With

        .ChartArea.Format.Line.ForeColor.RGB = RGB(180, 180, 180)
        .PlotArea.Format.Line.Visible = msoFalse

    End With

End Sub


Private Sub SetAutomaticYAxis( _
    ByVal chartAxis As Axis, _
    ByVal onRange As Range, _
    ByVal tenorRange As Range, _
    ByVal realizedRange As Range)

    Dim minimumValue As Double
    Dim maximumValue As Double
    Dim padding As Double

    minimumValue = Application.Min( _
        Application.Min(onRange), _
        Application.Min(tenorRange), _
        Application.Min(realizedRange))

    maximumValue = Application.Max( _
        Application.Max(onRange), _
        Application.Max(tenorRange), _
        Application.Max(realizedRange))

    padding = (maximumValue - minimumValue) * 0.15

    If padding < 0.001 Then
        padding = 0.001
    End If

    With chartAxis

        .MinimumScale = Application.Max(0, minimumValue - padding)
        .MaximumScale = maximumValue + padding

        If maximumValue - minimumValue < 0.01 Then
            .MajorUnit = 0.0025
        Else
            .MajorUnitIsAuto = True
        End If

    End With

End Sub


Private Function CompoundedONRate( _
    ByVal ws As Worksheet, _
    ByVal startDate As Date, _
    ByVal maturityDate As Date, _
    ByVal lastRawRow As Long) As Variant

    Dim rawRow As Long

    Dim rateDate As Date
    Dim nextRateDate As Date

    Dim accrualStart As Date
    Dim accrualEnd As Date

    Dim onRate As Double
    Dim dayCount As Long
    Dim totalDays As Long

    Dim growthFactor As Double
    Dim effectiveReturn As Double

    If Not DataAvailableThroughMaturity( _
        ws, maturityDate, lastRawRow) Then

        CompoundedONRate = CVErr(xlErrNA)
        Exit Function

    End If

    totalDays = DateDiff("d", startDate, maturityDate)

    If totalDays <= 0 Then

        CompoundedONRate = CVErr(xlErrValue)
        Exit Function

    End If

    growthFactor = 1#

    For rawRow = FIRST_RAW_ROW To lastRawRow

        If IsDate(ws.Cells(rawRow, RAW_DATE_COL).Value) And _
           IsNumeric(ws.Cells(rawRow, RAW_ON_COL).Value) Then

            rateDate = CDate(ws.Cells(rawRow, RAW_DATE_COL).Value)

            If rawRow < lastRawRow And _
               IsDate(ws.Cells(rawRow + 1, RAW_DATE_COL).Value) Then

                nextRateDate = _
                    CDate(ws.Cells(rawRow + 1, RAW_DATE_COL).Value)

            Else

                nextRateDate = maturityDate

            End If

            accrualStart = LaterDate(rateDate, startDate)
            accrualEnd = EarlierDate(nextRateDate, maturityDate)

            If accrualEnd > accrualStart Then

                dayCount = DateDiff("d", accrualStart, accrualEnd)

                onRate = _
                    CDbl(ws.Cells(rawRow, RAW_ON_COL).Value) / 100#

                growthFactor = growthFactor * _
                    (1# + onRate * dayCount / 360#)

            End If

            If rateDate >= maturityDate Then Exit For

        End If

    Next rawRow

    effectiveReturn = growthFactor - 1#

    'Annualized effective ON rate, expressed in percentage points.
    CompoundedONRate = _
        effectiveReturn * 360# / totalDays * 100#

End Function


Private Function GetRateForDate( _
    ByVal ws As Worksheet, _
    ByVal targetDate As Date, _
    ByVal rateColumn As Long, _
    ByVal lastRawRow As Long) As Variant

    Dim rawRow As Long
    Dim rateDate As Date
    Dim lastValidRate As Variant

    lastValidRate = CVErr(xlErrNA)

    For rawRow = FIRST_RAW_ROW To lastRawRow

        If IsDate(ws.Cells(rawRow, RAW_DATE_COL).Value) Then

            rateDate = CDate(ws.Cells(rawRow, RAW_DATE_COL).Value)

            If rateDate > targetDate Then Exit For

            If rateDate <= targetDate And _
               IsNumeric(ws.Cells(rawRow, rateColumn).Value) Then

                If CDbl(ws.Cells(rawRow, rateColumn).Value) <> 0 Then
                    lastValidRate = _
                        CDbl(ws.Cells(rawRow, rateColumn).Value)
                End If

            End If

        End If

    Next rawRow

    GetRateForDate = lastValidRate

End Function


Private Function DataAvailableThroughMaturity( _
    ByVal ws As Worksheet, _
    ByVal maturityDate As Date, _
    ByVal lastRawRow As Long) As Boolean

    If IsDate(ws.Cells(lastRawRow, RAW_DATE_COL).Value) Then

        DataAvailableThroughMaturity = _
            CDate(ws.Cells(lastRawRow, RAW_DATE_COL).Value) >= maturityDate

    Else

        DataAvailableThroughMaturity = False

    End If

End Function


Private Function LaterDate( _
    ByVal date1 As Date, _
    ByVal date2 As Date) As Date

    If date1 > date2 Then
        LaterDate = date1
    Else
        LaterDate = date2
    End If

End Function


Private Function EarlierDate( _
    ByVal date1 As Date, _
    ByVal date2 As Date) As Date

    If date1 < date2 Then
        EarlierDate = date1
    Else
        EarlierDate = date2
    End If

End Function


Private Function GetOrCreateOutputSheet( _
    ByVal sheetName As String) As Worksheet

    On Error Resume Next

    Set GetOrCreateOutputSheet = _
        ThisWorkbook.Worksheets(sheetName)

    On Error GoTo 0

    If GetOrCreateOutputSheet Is Nothing Then

        Set GetOrCreateOutputSheet = _
            ThisWorkbook.Worksheets.Add( _
                After:=ThisWorkbook.Worksheets( _
                    ThisWorkbook.Worksheets.Count))

        GetOrCreateOutputSheet.Name = sheetName

    End If

End Function


Private Sub ClearPreviousAnalysis(ByVal ws As Worksheet)

    Dim chartObject As chartObject

    ws.Cells.Clear

    For Each chartObject In ws.ChartObjects
        chartObject.Delete
    Next chartObject

End Sub


Private Sub CreateAnalysisHeader(ByVal ws As Worksheet)

    With ws.Range("A1:AF1")

        .Merge
        .Value = _
            "SOFR Term Strategy Analysis – Pricing at T0 vs Realized ON"

        .Font.Bold = True
        .Font.Size = 16
        .Font.Color = RGB(255, 255, 255)

        .Interior.Color = RGB(31, 78, 121)

        .HorizontalAlignment = xlCenter
        .VerticalAlignment = xlCenter

    End With

    ws.Rows(1).RowHeight = 28

End Sub


Private Sub FormatTenorTable( _
    ByVal ws As Worksheet, _
    ByVal startColumn As Long, _
    ByVal headerRow As Long, _
    ByVal firstDataRow As Long, _
    ByVal lastDataRow As Long)

    Dim tableRange As Range

    Set tableRange = ws.Range( _
        ws.Cells(headerRow, startColumn), _
        ws.Cells(lastDataRow, startColumn + 8))

    With tableRange.Borders

        .LineStyle = xlContinuous
        .Color = RGB(210, 210, 210)
        .Weight = xlThin

    End With

    ws.Range( _
        ws.Cells(firstDataRow, startColumn), _
        ws.Cells(lastDataRow, startColumn + 1)).NumberFormat = _
        "dd-mmm-yy"

    ws.Range( _
        ws.Cells(firstDataRow, startColumn + 2), _
        ws.Cells(lastDataRow, startColumn + 4)).NumberFormat = _
        "0.000%"

    ws.Range( _
        ws.Cells(firstDataRow, startColumn + 5), _
        ws.Cells(lastDataRow, startColumn + 6)).NumberFormat = _
        "0.0"

    ws.Range( _
        ws.Cells(firstDataRow, startColumn + 8), _
        ws.Cells(lastDataRow, startColumn + 8)).NumberFormat = _
        "0"

    ApplyOutcomeFormatting _
        ws:=ws, _
        outcomeRange:=ws.Range( _
            ws.Cells(firstDataRow, startColumn + 7), _
            ws.Cells(lastDataRow, startColumn + 7))

End Sub


Private Sub ApplyOutcomeFormatting( _
    ByVal ws As Worksheet, _
    ByVal outcomeRange As Range)

    outcomeRange.FormatConditions.Delete

    With outcomeRange.FormatConditions.Add( _
        Type:=xlTextString, _
        String:="Outperformed", _
        TextOperator:=xlContains)

        .Interior.Color = RGB(221, 235, 247)
        .Font.Color = RGB(31, 78, 121)

    End With

    With outcomeRange.FormatConditions.Add( _
        Type:=xlTextString, _
        String:="ON Outperformed", _
        TextOperator:=xlContains)

        .Interior.Color = RGB(252, 228, 214)
        .Font.Color = RGB(192, 0, 0)

    End With

End Sub


Private Sub FormatOutputSheet(ByVal ws As Worksheet)

    ws.Columns("A:B").ColumnWidth = 12
    ws.Columns("C:E").ColumnWidth = 12
    ws.Columns("F:G").ColumnWidth = 15
    ws.Columns("H").ColumnWidth = 19
    ws.Columns("I").ColumnWidth = 9

    ws.Columns("Q:R").ColumnWidth = 12
    ws.Columns("S:U").ColumnWidth = 12
    ws.Columns("V:W").ColumnWidth = 15
    ws.Columns("X").ColumnWidth = 19
    ws.Columns("Y").ColumnWidth = 9

    ws.Rows("3:60").RowHeight = 18

    ws.Activate
    ActiveWindow.DisplayGridlines = False

End Sub
