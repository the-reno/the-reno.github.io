Attribute VB_Name = "DailyTenorPositionView"
Option Explicit

' Daily tenor position view.
' Copy this code into a standard VBA module, then run BuildDailyPositionTemplate.
'
' Purpose:
' - Creates a calendar-day view for each tenor across the selected period.
' - Shows the decision/action taken each day:
'   tenor, calendar date, position date, initial cash, final cash,
'   rate used, interest earned, rolling date TRUE/FALSE.
'
' Required curve columns on Curve sheet:
' date, ON, 1M, 2M, 3M, 6M
'
' User inputs on Inputs sheet:
' B4 = last date
' B5 = lookback years, e.g. 1, 1.5, 2
' B6 = notional

Private Type RatePoint
    CurveDate As Date
    ONRate As Double
    M1Rate As Double
    M2Rate As Double
    M3Rate As Double
    M6Rate As Double
End Type

Private Type RollPoint
    Tenor As String
    RollNo As Long
    StartDate As Date
    RateDate As Date
    MaturityTargetDate As Date
    ActualEndDate As Date
    RateUsed As Double
    AccrualDays As Long
    StartingCash As Double
    InterestEarned As Double
    EndingCash As Double
End Type

Private Rates() As RatePoint
Private RateCount As Long
Private Rolls() As RollPoint
Private RollCount As Long

Public Sub BuildDailyPositionTemplate()
    Application.ScreenUpdating = False
    Application.DisplayAlerts = False
    EnsureSheet "Inputs", False
    EnsureSheet "Curve", False
    EnsureSheet "Daily_Position", True
    EnsureSheet "Roll_Schedule", True
    If WorksheetFunction.CountA(ThisWorkbook.Worksheets("Inputs").Cells) = 0 Then BuildDefaultInputs
    If WorksheetFunction.CountA(ThisWorkbook.Worksheets("Curve").Cells) = 0 Then BuildSampleCurve
    Application.DisplayAlerts = True
    Application.ScreenUpdating = True
    RunDailyPositionView
End Sub

Public Sub RunDailyPositionView()
    Dim wsIn As Worksheet, wsDaily As Worksheet, wsRoll As Worksheet
    Dim lastDateInput As Date, lookbackYears As Double, notional As Double
    Dim targetDay As Long, lookbackMonths As Long
    Dim targetStartDate As Date, actualStartDate As Date, actualEndDate As Date
    Dim tenor As Variant

    Set wsIn = ThisWorkbook.Worksheets("Inputs")
    Set wsDaily = ThisWorkbook.Worksheets("Daily_Position")
    Set wsRoll = ThisWorkbook.Worksheets("Roll_Schedule")
    lastDateInput = CDate(wsIn.Range("B4").Value)
    lookbackYears = CDbl(wsIn.Range("B5").Value)
    notional = CDbl(wsIn.Range("B6").Value)
    targetDay = Day(lastDateInput)
    lookbackMonths = CLng(Round(lookbackYears * 12#, 0))

    LoadRatesFromCurve
    actualEndDate = PreviousCurveDate(lastDateInput)
    targetStartDate = AddMonthsPreserveDay(lastDateInput, -lookbackMonths, targetDay)
    actualStartDate = PreviousCurveDate(targetStartDate)
    wsIn.Range("B8").Value = targetStartDate
    wsIn.Range("B9").Value = actualStartDate
    wsIn.Range("B10").Value = actualEndDate
    wsIn.Range("B11").Value = targetDay
    wsIn.Range("B12").Value = Now

    RollCount = 0
    Erase Rolls
    For Each tenor In Array("ON", "1M", "2M", "3M", "6M")
        SimulateTenor CStr(tenor), actualStartDate, targetStartDate, actualEndDate, lastDateInput, notional, targetDay
    Next tenor
    WriteRollSchedule wsRoll
    WriteDailyPosition wsDaily, actualStartDate, actualEndDate
    FormatOutputs wsIn, wsDaily, wsRoll
    wsDaily.Activate
End Sub

Private Sub EnsureSheet(ByVal sheetName As String, ByVal clearSheet As Boolean)
    Dim ws As Worksheet
    On Error Resume Next
    Set ws = ThisWorkbook.Worksheets(sheetName)
    On Error GoTo 0
    If ws Is Nothing Then
        Set ws = ThisWorkbook.Worksheets.Add(After:=ThisWorkbook.Worksheets(ThisWorkbook.Worksheets.Count))
        ws.Name = sheetName
    ElseIf clearSheet Then
        ws.Cells.Clear
    End If
End Sub

Private Sub BuildDefaultInputs()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Worksheets("Inputs")
    ws.Cells.Clear
    ws.Range("A1").Value = "Daily tenor position view"
    ws.Range("A1").Font.Bold = True
    ws.Range("A3").Value = "Input": ws.Range("B3").Value = "Value"
    ws.Range("A4").Value = "Last date": ws.Range("B4").Value = DateSerial(2026, 6, 30)
    ws.Range("A5").Value = "Lookback years": ws.Range("B5").Value = 1
    ws.Range("A6").Value = "Notional": ws.Range("B6").Value = 1000000
    ws.Range("A8").Value = "Calculated target start date"
    ws.Range("A9").Value = "Calculated actual start date"
    ws.Range("A10").Value = "Calculated common end date"
    ws.Range("A11").Value = "Target day of month"
    ws.Range("A12").Value = "Last run"
    ws.Range("D3").Value = "Notes"
    ws.Range("D4").Value = "Daily_Position includes all calendar dates."
    ws.Range("D5").Value = "Rolling date TRUE means a new investment decision starts that day."
    ws.Range("D6").Value = "Curve dates use previous available date when needed."
    ws.Columns.AutoFit
End Sub

Private Sub BuildSampleCurve()
    Dim ws As Worksheet, d As Date, rowNum As Long, t As Double, baseRate As Double
    Set ws = ThisWorkbook.Worksheets("Curve")
    ws.Cells.Clear
    ws.Range("A1:F1").Value = Array("date", "ON", "1M", "2M", "3M", "6M")
    ws.Rows(1).Font.Bold = True
    rowNum = 2
    For d = DateSerial(2021, 1, 4) To DateSerial(2026, 6, 30)
        If Weekday(d, vbMonday) <= 5 Then
            t = DateDiff("d", DateSerial(2021, 1, 4), d) / 365#
            baseRate = 0.008 + WorksheetFunction.Min(0.044, 0.008 * t) + 0.004 * Sin(t * 2.1)
            ws.Cells(rowNum, 1).Value = d
            ws.Cells(rowNum, 2).Value = WorksheetFunction.Max(0.0005, baseRate - 0.0015)
            ws.Cells(rowNum, 3).Value = WorksheetFunction.Max(0.0005, baseRate - 0.0005)
            ws.Cells(rowNum, 4).Value = WorksheetFunction.Max(0.0005, baseRate + 0.0002)
            ws.Cells(rowNum, 5).Value = WorksheetFunction.Max(0.0005, baseRate + 0.0008)
            ws.Cells(rowNum, 6).Value = WorksheetFunction.Max(0.0005, baseRate + 0.0018)
            rowNum = rowNum + 1
        End If
    Next d
End Sub

Private Sub LoadRatesFromCurve()
    Dim ws As Worksheet, lastRow As Long, r As Long
    Set ws = ThisWorkbook.Worksheets("Curve")
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row
    If lastRow < 2 Then Err.Raise vbObjectError + 100, , "Curve sheet has no rate data."
    RateCount = lastRow - 1
    ReDim Rates(1 To RateCount)
    For r = 2 To lastRow
        Rates(r - 1).CurveDate = CDate(ws.Cells(r, 1).Value)
        Rates(r - 1).ONRate = NormalizeRate(ws.Cells(r, 2).Value)
        Rates(r - 1).M1Rate = NormalizeRate(ws.Cells(r, 3).Value)
        Rates(r - 1).M2Rate = NormalizeRate(ws.Cells(r, 4).Value)
        Rates(r - 1).M3Rate = NormalizeRate(ws.Cells(r, 5).Value)
        Rates(r - 1).M6Rate = NormalizeRate(ws.Cells(r, 6).Value)
    Next r
    SortRatesByDate
End Sub

Private Function NormalizeRate(ByVal v As Variant) As Double
    NormalizeRate = CDbl(v)
    If Abs(NormalizeRate) > 1# Then NormalizeRate = NormalizeRate / 100#
End Function

Private Sub SortRatesByDate()
    Dim i As Long, j As Long, tmp As RatePoint
    For i = 1 To RateCount - 1
        For j = i + 1 To RateCount
            If Rates(j).CurveDate < Rates(i).CurveDate Then
                tmp = Rates(i): Rates(i) = Rates(j): Rates(j) = tmp
            End If
        Next j
    Next i
End Sub

Private Sub SimulateTenor(ByVal tenor As String, ByVal actualStartDate As Date, ByVal targetStartDate As Date, ByVal actualEndDate As Date, ByVal lastDate As Date, ByVal notional As Double, ByVal targetDay As Long)
    Dim cash As Double, investStart As Date, rollNo As Long
    Dim maturityTarget As Date, actualEnd As Date, rateDate As Date, rateUsed As Double
    Dim targets As Collection, i As Long
    cash = notional
    investStart = actualStartDate
    rollNo = 1
    If tenor = "ON" Then
        Do While investStart < actualEndDate
            rateDate = PreviousCurveDate(investStart)
            rateUsed = GetRateForTenor(rateDate, tenor)
            actualEnd = NextCurveDate(investStart)
            If actualEnd > actualEndDate Then actualEnd = actualEndDate
            AddRoll tenor, rollNo, investStart, rateDate, actualEnd, actualEnd, rateUsed, cash
            cash = Rolls(RollCount).EndingCash
            investStart = actualEnd
            rollNo = rollNo + 1
        Loop
        Exit Sub
    End If
    Set targets = BuildTenorTargets(targetStartDate, lastDate, TenorMonths(tenor), targetDay)
    For i = 2 To targets.Count
        If investStart >= actualEndDate Then Exit For
        maturityTarget = CDate(targets(i))
        actualEnd = PreviousCurveDate(maturityTarget)
        If actualEnd <= investStart Then actualEnd = NextCurveDate(investStart)
        If actualEnd > actualEndDate Then actualEnd = actualEndDate
        rateDate = PreviousCurveDate(investStart)
        rateUsed = GetRateForTenor(rateDate, tenor)
        AddRoll tenor, rollNo, investStart, rateDate, maturityTarget, actualEnd, rateUsed, cash
        cash = Rolls(RollCount).EndingCash
        investStart = actualEnd
        rollNo = rollNo + 1
    Next i
End Sub

Private Sub AddRoll(ByVal tenor As String, ByVal rollNo As Long, ByVal startDate As Date, ByVal rateDate As Date, ByVal maturityTarget As Date, ByVal actualEnd As Date, ByVal rateUsed As Double, ByVal cash As Double)
    Dim days As Long, interest As Double
    RollCount = RollCount + 1
    If RollCount = 1 Then ReDim Rolls(1 To 1) Else ReDim Preserve Rolls(1 To RollCount)
    days = DateDiff("d", startDate, actualEnd)
    interest = cash * rateUsed * days / 360#
    With Rolls(RollCount)
        .Tenor = tenor
        .RollNo = rollNo
        .StartDate = startDate
        .RateDate = rateDate
        .MaturityTargetDate = maturityTarget
        .ActualEndDate = actualEnd
        .RateUsed = rateUsed
        .AccrualDays = days
        .StartingCash = cash
        .InterestEarned = interest
        .EndingCash = cash + interest
    End With
End Sub

Private Sub WriteRollSchedule(ByVal ws As Worksheet)
    Dim i As Long
    ws.Cells.Clear
    ws.Range("A1:K1").Value = Array("tenor", "roll number", "investment start date", "rate date used", "maturity target date", "actual end date", "rate used", "accrual days", "initial cash", "interest earned", "final cash")
    For i = 1 To RollCount
        ws.Cells(i + 1, 1).Value = Rolls(i).Tenor
        ws.Cells(i + 1, 2).Value = Rolls(i).RollNo
        ws.Cells(i + 1, 3).Value = Rolls(i).StartDate
        ws.Cells(i + 1, 4).Value = Rolls(i).RateDate
        ws.Cells(i + 1, 5).Value = Rolls(i).MaturityTargetDate
        ws.Cells(i + 1, 6).Value = Rolls(i).ActualEndDate
        ws.Cells(i + 1, 7).Value = Rolls(i).RateUsed
        ws.Cells(i + 1, 8).Value = Rolls(i).AccrualDays
        ws.Cells(i + 1, 9).Value = Rolls(i).StartingCash
        ws.Cells(i + 1, 10).Value = Rolls(i).InterestEarned
        ws.Cells(i + 1, 11).Value = Rolls(i).EndingCash
    Next i
End Sub

Private Sub WriteDailyPosition(ByVal ws As Worksheet, ByVal actualStartDate As Date, ByVal actualEndDate As Date)
    Dim rowNum As Long, d As Date, tenor As Variant, rp As RollPoint
    Dim isInvested As Boolean, isRollingDate As Boolean
    Dim initialCash As Variant, finalCash As Variant, rateUsed As Variant, interestEarned As Variant
    Dim decision As String
    ws.Cells.Clear
    ws.Range("A1:L1").Value = Array("calendar date", "tenor", "position date", "initial cash", "final cash", "rate used", "interest earned", "rolling date", "decision", "active roll number", "rate date used", "maturity target date")
    rowNum = 2
    For d = actualStartDate To actualEndDate
        For Each tenor In Array("ON", "1M", "2M", "3M", "6M")
            If FindRollForCalendarDate(CStr(tenor), d, rp, isInvested, isRollingDate) Then
                initialCash = DailyInitialCash(rp, d, isRollingDate)
                finalCash = DailyFinalCash(rp, d)
                rateUsed = rp.RateUsed
                interestEarned = DailyInterestEarned(rp, d, isInvested)
                If isRollingDate Then
                    decision = "Roll/start investment using rate from " & Format$(rp.RateDate, "yyyy-mm-dd")
                ElseIf d = rp.ActualEndDate Then
                    decision = "Maturity/cash available"
                ElseIf isInvested Then
                    decision = "Hold existing investment"
                Else
                    decision = "No active position"
                End If
                ws.Cells(rowNum, 1).Value = d
                ws.Cells(rowNum, 2).Value = CStr(tenor)
                ws.Cells(rowNum, 3).Value = PreviousCurveDate(d)
                ws.Cells(rowNum, 4).Value = initialCash
                ws.Cells(rowNum, 5).Value = finalCash
                ws.Cells(rowNum, 6).Value = rateUsed
                ws.Cells(rowNum, 7).Value = interestEarned
                ws.Cells(rowNum, 8).Value = isRollingDate
                ws.Cells(rowNum, 9).Value = decision
                ws.Cells(rowNum, 10).Value = rp.RollNo
                ws.Cells(rowNum, 11).Value = rp.RateDate
                ws.Cells(rowNum, 12).Value = rp.MaturityTargetDate
            End If
            rowNum = rowNum + 1
        Next tenor
    Next d
End Sub

Private Function FindRollForCalendarDate(ByVal tenor As String, ByVal calendarDate As Date, ByRef outRoll As RollPoint, ByRef isInvested As Boolean, ByRef isRollingDate As Boolean) As Boolean
    Dim i As Long, lastCompleted As Long
    isInvested = False
    isRollingDate = False
    For i = 1 To RollCount
        If Rolls(i).Tenor = tenor And Rolls(i).StartDate = calendarDate Then
            outRoll = Rolls(i)
            isInvested = (calendarDate < Rolls(i).ActualEndDate)
            isRollingDate = True
            FindRollForCalendarDate = True
            Exit Function
        End If
    Next i
    For i = 1 To RollCount
        If Rolls(i).Tenor = tenor Then
            If calendarDate >= Rolls(i).StartDate And calendarDate <= Rolls(i).ActualEndDate Then
                outRoll = Rolls(i)
                isInvested = (calendarDate < Rolls(i).ActualEndDate)
                isRollingDate = (calendarDate = Rolls(i).StartDate)
                FindRollForCalendarDate = True
                Exit Function
            End If
            If Rolls(i).ActualEndDate < calendarDate Then lastCompleted = i
        End If
    Next i
    If lastCompleted > 0 Then
        outRoll = Rolls(lastCompleted)
        FindRollForCalendarDate = True
    End If
End Function

Private Function DailyInitialCash(ByVal rp As RollPoint, ByVal calendarDate As Date, ByVal isRollingDate As Boolean) As Variant
    If calendarDate > rp.ActualEndDate Then
        DailyInitialCash = rp.EndingCash
    Else
        DailyInitialCash = rp.StartingCash
    End If
End Function

Private Function DailyFinalCash(ByVal rp As RollPoint, ByVal calendarDate As Date) As Variant
    If calendarDate >= rp.ActualEndDate Then
        DailyFinalCash = rp.EndingCash
    Else
        DailyFinalCash = rp.StartingCash + rp.StartingCash * rp.RateUsed * DateDiff("d", rp.StartDate, calendarDate) / 360#
    End If
End Function

Private Function DailyInterestEarned(ByVal rp As RollPoint, ByVal calendarDate As Date, ByVal isInvested As Boolean) As Variant
    If calendarDate < rp.StartDate Then
        DailyInterestEarned = 0#
    ElseIf calendarDate >= rp.ActualEndDate Then
        DailyInterestEarned = rp.InterestEarned
    ElseIf isInvested Then
        DailyInterestEarned = rp.StartingCash * rp.RateUsed * DateDiff("d", rp.StartDate, calendarDate) / 360#
    Else
        DailyInterestEarned = 0#
    End If
End Function

Private Function BuildTenorTargets(ByVal startTarget As Date, ByVal endTarget As Date, ByVal monthsStep As Long, ByVal targetDay As Long) As Collection
    Dim reverseTargets As New Collection, currentDate As Date, i As Long
    Set BuildTenorTargets = New Collection
    currentDate = endTarget
    Do While currentDate >= startTarget
        reverseTargets.Add currentDate
        currentDate = AddMonthsPreserveDay(currentDate, -monthsStep, targetDay)
    Loop
    For i = reverseTargets.Count To 1 Step -1
        BuildTenorTargets.Add reverseTargets(i)
    Next i
End Function

Private Function AddMonthsPreserveDay(ByVal sourceDate As Date, ByVal months As Long, ByVal targetDay As Long) As Date
    Dim y As Long, m As Long, lastDay As Long
    y = Year(sourceDate)
    m = Month(sourceDate) + months
    Do While m > 12
        y = y + 1
        m = m - 12
    Loop
    Do While m < 1
        y = y - 1
        m = m + 12
    Loop
    lastDay = Day(DateSerial(y, m + 1, 0))
    AddMonthsPreserveDay = DateSerial(y, m, WorksheetFunction.Min(targetDay, lastDay))
End Function

Private Function PreviousCurveDate(ByVal targetDate As Date) As Date
    Dim i As Long
    For i = RateCount To 1 Step -1
        If Rates(i).CurveDate <= targetDate Then
            PreviousCurveDate = Rates(i).CurveDate
            Exit Function
        End If
    Next i
    Err.Raise vbObjectError + 101, , "No curve date exists on or before " & Format$(targetDate, "yyyy-mm-dd")
End Function

Private Function NextCurveDate(ByVal targetDate As Date) As Date
    Dim i As Long
    For i = 1 To RateCount
        If Rates(i).CurveDate > targetDate Then
            NextCurveDate = Rates(i).CurveDate
            Exit Function
        End If
    Next i
    NextCurveDate = Rates(RateCount).CurveDate
End Function

Private Function GetRateForTenor(ByVal rateDate As Date, ByVal tenor As String) As Double
    Dim i As Long
    For i = 1 To RateCount
        If Rates(i).CurveDate = rateDate Then
            Select Case tenor
                Case "ON": GetRateForTenor = Rates(i).ONRate
                Case "1M": GetRateForTenor = Rates(i).M1Rate
                Case "2M": GetRateForTenor = Rates(i).M2Rate
                Case "3M": GetRateForTenor = Rates(i).M3Rate
                Case "6M": GetRateForTenor = Rates(i).M6Rate
                Case Else: Err.Raise vbObjectError + 102, , "Unsupported tenor: " & tenor
            End Select
            Exit Function
        End If
    Next i
    Err.Raise vbObjectError + 103, , "Rate date not found: " & Format$(rateDate, "yyyy-mm-dd")
End Function

Private Function TenorMonths(ByVal tenor As String) As Long
    Select Case tenor
        Case "1M": TenorMonths = 1
        Case "2M": TenorMonths = 2
        Case "3M": TenorMonths = 3
        Case "6M": TenorMonths = 6
        Case Else: TenorMonths = 0
    End Select
End Function

Private Sub FormatOutputs(ByVal wsIn As Worksheet, ByVal wsDaily As Worksheet, ByVal wsRoll As Worksheet)
    With wsIn
        .Range("B4").NumberFormat = "yyyy-mm-dd"
        .Range("B5").NumberFormat = "0.00"
        .Range("B6").NumberFormat = "$#,##0.00"
        .Range("B8:B10").NumberFormat = "yyyy-mm-dd"
        .Columns.AutoFit
    End With
    With wsDaily
        .Rows(1).Font.Bold = True
        .Columns("A:A").NumberFormat = "yyyy-mm-dd"
        .Columns("C:C").NumberFormat = "yyyy-mm-dd"
        .Columns("D:E").NumberFormat = "$#,##0.00"
        .Columns("F:F").NumberFormat = "0.0000%"
        .Columns("G:G").NumberFormat = "$#,##0.00"
        .Columns("K:L").NumberFormat = "yyyy-mm-dd"
        .Columns.AutoFit
    End With
    With wsRoll
        .Rows(1).Font.Bold = True
        .Columns("C:F").NumberFormat = "yyyy-mm-dd"
        .Columns("G:G").NumberFormat = "0.0000%"
        .Columns("I:K").NumberFormat = "$#,##0.00"
        .Columns.AutoFit
    End With
End Sub
