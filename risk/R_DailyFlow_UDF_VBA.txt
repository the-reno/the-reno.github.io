Attribute VB_Name = "RDailyFlowUDF"
Option Explicit

' Worksheet formula:
'   =r_dailyflow(last_date, years, amount, curve_range)
'
' Curve columns:
'   Date | ON | 1M | 2M | 3M | 6M
'
' Roll convention:
'   1. last_date is the exact common final maturity date.
'   2. For fixed tenors, each prior target is calculated backward
'      from the following maturity date.
'   3. Actual roll date = latest curve date on or before the target.
'   4. The lookback boundary is not automatically a roll date.
'      The first retained roll is the first scheduled roll on or after
'      the requested lookback boundary.
'   5. Roll flag is TRUE only on an actual investment/reinvestment date.
'      The final maturity date is not flagged as a new roll.
'   6. ON rolls on each curve date before the final maturity date.
'
' Rates may be stored as 3.55 or 3.55%.
' The function normalizes both forms to decimal rates internally.

Private Type RatePoint
    CurveDate As Date
    ONRate As Double
    M1Rate As Double
    M2Rate As Double
    M3Rate As Double
    M6Rate As Double
End Type

Private Type RollPoint
    StartDate As Date
    MaturityDate As Date
    TargetMaturityDate As Date
    RateDate As Date
    RateUsed As Double
    StartCash As Double
    EndCash As Double
    RollNo As Long
End Type

Private UdfRates() As RatePoint
Private UdfRateCount As Long

Public Function r_dailyflow( _
    ByVal last_date As Variant, _
    ByVal years As Double, _
    ByVal amount As Double, _
    ByVal curve As Range) As Variant

    On Error GoTo Fail

    Dim analysisEnd As Date
    Dim requestedStart As Date
    Dim analysisStart As Date
    Dim targetDay As Long
    Dim lookbackMonths As Long
    Dim totalDays As Long
    Dim result() As Variant
    Dim schedules(1 To 5) As Variant
    Dim tenorNames(1 To 5) As String
    Dim d As Date
    Dim dayRow As Long
    Dim tenorIndex As Long
    Dim outputCol As Long
    Dim rp As RollPoint
    Dim hasPosition As Boolean
    Dim isRoll As Boolean

    If Not IsDate(last_date) And Not IsNumeric(last_date) Then
        Err.Raise vbObjectError + 1, , "Invalid analysis date."
    End If

    If years <= 0 Then
        Err.Raise vbObjectError + 2, , "Lookback years must be greater than zero."
    End If

    If amount <= 0 Then
        Err.Raise vbObjectError + 3, , "Initial cash must be greater than zero."
    End If

    LoadRatesFromRange curve

    analysisEnd = ToDate(last_date)

    If Not IsExactCurveDate(analysisEnd) Then
        Err.Raise vbObjectError + 4, , _
            "The analysis date must exist exactly in the curve: " & _
            Format$(analysisEnd, "yyyy-mm-dd")
    End If

    targetDay = Day(analysisEnd)
    lookbackMonths = CLng(Round(years * 12#, 0))
    requestedStart = AddMonthsPreserveDay(analysisEnd, -lookbackMonths, targetDay)

    tenorNames(1) = "ON"
    tenorNames(2) = "1M"
    tenorNames(3) = "2M"
    tenorNames(4) = "3M"
    tenorNames(5) = "6M"

    schedules(1) = BuildONSchedule(requestedStart, analysisEnd, amount)
    schedules(2) = BuildFixedSchedule(requestedStart, analysisEnd, 1, "1M", amount, targetDay)
    schedules(3) = BuildFixedSchedule(requestedStart, analysisEnd, 2, "2M", amount, targetDay)
    schedules(4) = BuildFixedSchedule(requestedStart, analysisEnd, 3, "3M", amount, targetDay)
    schedules(5) = BuildFixedSchedule(requestedStart, analysisEnd, 6, "6M", amount, targetDay)

    analysisStart = EarliestScheduleStart(schedules)

    If analysisStart = 0 Then
        Err.Raise vbObjectError + 5, , "No valid roll schedule exists inside the requested period."
    End If

    totalDays = DateDiff("d", analysisStart, analysisEnd) + 1
    ReDim result(1 To totalDays + 1, 1 To 26)

    result(1, 1) = "Date"
    outputCol = 2

    For tenorIndex = 1 To 5
        result(1, outputCol) = tenorNames(tenorIndex) & " Rate"
        result(1, outputCol + 1) = tenorNames(tenorIndex) & " Maturity"
        result(1, outputCol + 2) = tenorNames(tenorIndex) & " Interest Earned"
        result(1, outputCol + 3) = tenorNames(tenorIndex) & " Value"
        result(1, outputCol + 4) = tenorNames(tenorIndex) & " Roll"
        outputCol = outputCol + 5
    Next tenorIndex

    dayRow = 2

    For d = analysisStart To analysisEnd
        result(dayRow, 1) = d
        outputCol = 2

        For tenorIndex = 1 To 5
            hasPosition = FindSchedulePosition(schedules(tenorIndex), d, rp, isRoll)

            If hasPosition Then
                result(dayRow, outputCol) = rp.RateUsed
                result(dayRow, outputCol + 1) = rp.MaturityDate
                result(dayRow, outputCol + 2) = PositionInterest(rp, d)
                result(dayRow, outputCol + 3) = PositionValue(rp, d)
                result(dayRow, outputCol + 4) = isRoll
            Else
                result(dayRow, outputCol) = Empty
                result(dayRow, outputCol + 1) = Empty
                result(dayRow, outputCol + 2) = 0#
                result(dayRow, outputCol + 3) = 0#
                result(dayRow, outputCol + 4) = False
            End If

            outputCol = outputCol + 5
        Next tenorIndex

        dayRow = dayRow + 1
    Next d

    r_dailyflow = result
    Exit Function

Fail:
    r_dailyflow = "r_dailyflow error: " & Err.Description
End Function

Private Sub LoadRatesFromRange(ByVal curve As Range)
    Dim dataRange As Range
    Dim data As Variant
    Dim firstRow As Long
    Dim lastRow As Long
    Dim startRow As Long
    Dim i As Long
    Dim outIdx As Long
    Dim dateCol As Long
    Dim onCol As Long
    Dim m1Col As Long
    Dim m2Col As Long
    Dim m3Col As Long
    Dim m6Col As Long

    If curve.Columns.Count < 6 Then
        Err.Raise vbObjectError + 10, , _
            "Curve range must include Date, ON, 1M, 2M, 3M and 6M."
    End If

    firstRow = curve.Row
    lastRow = curve.Worksheet.Cells( _
        curve.Worksheet.Rows.Count, curve.Column).End(xlUp).Row

    Set dataRange = curve.Worksheet.Range( _
        curve.Worksheet.Cells(firstRow, curve.Column), _
        curve.Worksheet.Cells(lastRow, curve.Column + 5))

    data = dataRange.Value2

    If IsDateValue(data(1, 1)) Then
        startRow = 1
        dateCol = 1
        onCol = 2
        m1Col = 3
        m2Col = 4
        m3Col = 5
        m6Col = 6
    Else
        startRow = 2
        dateCol = FindHeaderColumn(data, "date")
        onCol = FindHeaderColumn(data, "ON")
        m1Col = FindHeaderColumn(data, "1M")
        m2Col = FindHeaderColumn(data, "2M")
        m3Col = FindHeaderColumn(data, "3M")
        m6Col = FindHeaderColumn(data, "6M")
    End If

    ReDim UdfRates(1 To UBound(data, 1) - startRow + 1)
    outIdx = 0

    For i = startRow To UBound(data, 1)
        If Len(Trim$(CStr(data(i, dateCol)))) > 0 Then
            outIdx = outIdx + 1
            UdfRates(outIdx).CurveDate = ToDate(data(i, dateCol))
            UdfRates(outIdx).ONRate = NormalizeRate(data(i, onCol))
            UdfRates(outIdx).M1Rate = NormalizeRate(data(i, m1Col))
            UdfRates(outIdx).M2Rate = NormalizeRate(data(i, m2Col))
            UdfRates(outIdx).M3Rate = NormalizeRate(data(i, m3Col))
            UdfRates(outIdx).M6Rate = NormalizeRate(data(i, m6Col))
        End If
    Next i

    UdfRateCount = outIdx

    If UdfRateCount = 0 Then
        Err.Raise vbObjectError + 11, , "Curve contains no valid rate rows."
    End If

    ReDim Preserve UdfRates(1 To UdfRateCount)
    SortRatesByDate
End Sub

Private Function BuildONSchedule( _
    ByVal requestedStart As Date, _
    ByVal analysisEnd As Date, _
    ByVal amount As Double) As Variant

    Dim temp() As RollPoint
    Dim count As Long
    Dim i As Long
    Dim startDate As Date
    Dim endDate As Date
    Dim rateDate As Date
    Dim rateUsed As Double
    Dim cash As Double

    cash = amount
    count = 0

    For i = 1 To UdfRateCount - 1
        startDate = UdfRates(i).CurveDate

        If startDate >= requestedStart And startDate < analysisEnd Then
            endDate = UdfRates(i + 1).CurveDate
            If endDate > analysisEnd Then endDate = analysisEnd

            rateDate = startDate
            rateUsed = GetRateForTenor(rateDate, "ON")

            count = count + 1
            ReDim Preserve temp(1 To count)

            With temp(count)
                .StartDate = startDate
                .MaturityDate = endDate
                .TargetMaturityDate = endDate
                .RateDate = rateDate
                .RateUsed = rateUsed
                .StartCash = cash
                .EndCash = cash + cash * rateUsed * DateDiff("d", startDate, endDate) / 360#
                .RollNo = count
            End With

            cash = temp(count).EndCash

            If endDate = analysisEnd Then Exit For
        End If
    Next i

    If count = 0 Then
        BuildONSchedule = Empty
    Else
        BuildONSchedule = temp
    End If
End Function

Private Function BuildFixedSchedule( _
    ByVal requestedStart As Date, _
    ByVal analysisEnd As Date, _
    ByVal monthsStep As Long, _
    ByVal tenor As String, _
    ByVal amount As Double, _
    ByVal targetDay As Long) As Variant

    Const MAX_ROLLS As Long = 2000

    Dim backwardDates(1 To MAX_ROLLS) As Date
    Dim backwardTargets(1 To MAX_ROLLS) As Date
    Dim retainedDates() As Date
    Dim retainedTargets() As Date
    Dim schedule() As RollPoint
    Dim currentMaturity As Date
    Dim targetRoll As Date
    Dim actualRoll As Date
    Dim count As Long
    Dim keepCount As Long
    Dim i As Long
    Dim sourceIndex As Long
    Dim cash As Double
    Dim maturityDate As Date
    Dim targetMaturity As Date
    Dim rateUsed As Double

    currentMaturity = analysisEnd
    count = 0

    Do
        targetRoll = AddMonthsPreserveDay(currentMaturity, -monthsStep, targetDay)
        actualRoll = PreviousCurveDate(targetRoll)

        If actualRoll >= currentMaturity Then Exit Do

        count = count + 1
        backwardDates(count) = actualRoll
        backwardTargets(count) = currentMaturity

        currentMaturity = actualRoll

        If actualRoll < requestedStart Then Exit Do
        If count >= MAX_ROLLS Then Exit Do
    Loop

    keepCount = 0

    For i = count To 1 Step -1
        If backwardDates(i) >= requestedStart Then
            keepCount = keepCount + 1
            ReDim Preserve retainedDates(1 To keepCount)
            ReDim Preserve retainedTargets(1 To keepCount)
            retainedDates(keepCount) = backwardDates(i)
            retainedTargets(keepCount) = backwardTargets(i)
        End If
    Next i

    If keepCount = 0 Then
        BuildFixedSchedule = Empty
        Exit Function
    End If

    ReDim schedule(1 To keepCount)
    cash = amount

    For i = 1 To keepCount
        If i < keepCount Then
            maturityDate = retainedDates(i + 1)
        Else
            maturityDate = analysisEnd
        End If

        targetMaturity = retainedTargets(i)
        rateUsed = GetRateForTenor(retainedDates(i), tenor)

        With schedule(i)
            .StartDate = retainedDates(i)
            .MaturityDate = maturityDate
            .TargetMaturityDate = targetMaturity
            .RateDate = retainedDates(i)
            .RateUsed = rateUsed
            .StartCash = cash
            .EndCash = cash + cash * rateUsed * DateDiff("d", retainedDates(i), maturityDate) / 360#
            .RollNo = i
        End With

        cash = schedule(i).EndCash
    Next i

    BuildFixedSchedule = schedule
End Function

Private Function FindSchedulePosition( _
    ByVal schedule As Variant, _
    ByVal calendarDate As Date, _
    ByRef outRoll As RollPoint, _
    ByRef isRoll As Boolean) As Boolean

    Dim i As Long

    FindSchedulePosition = False
    isRoll = False

    If IsEmpty(schedule) Then Exit Function

    For i = LBound(schedule) To UBound(schedule)
        If calendarDate >= schedule(i).StartDate And _
           calendarDate <= schedule(i).MaturityDate Then

            outRoll = schedule(i)

            'A start date is a roll only when it opens a position.
            'The common final maturity date is never a new roll.
            isRoll = (calendarDate = schedule(i).StartDate And _
                      calendarDate < schedule(i).MaturityDate)

            FindSchedulePosition = True
            Exit Function
        End If
    Next i
End Function

Private Function PositionInterest(ByRef rp As RollPoint, ByVal calendarDate As Date) As Double
    Dim accrualEnd As Date

    If calendarDate <= rp.StartDate Then
        PositionInterest = 0#
        Exit Function
    End If

    accrualEnd = calendarDate
    If accrualEnd > rp.MaturityDate Then accrualEnd = rp.MaturityDate

    PositionInterest = rp.StartCash * rp.RateUsed * _
        DateDiff("d", rp.StartDate, accrualEnd) / 360#
End Function

Private Function PositionValue(ByRef rp As RollPoint, ByVal calendarDate As Date) As Double
    If calendarDate >= rp.MaturityDate Then
        PositionValue = rp.EndCash
    Else
        PositionValue = rp.StartCash + PositionInterest(rp, calendarDate)
    End If
End Function

Private Function EarliestScheduleStart(ByRef schedules() As Variant) As Date
    Dim i As Long
    Dim candidate As Date

    EarliestScheduleStart = 0

    For i = LBound(schedules) To UBound(schedules)
        If Not IsEmpty(schedules(i)) Then
            candidate = schedules(i)(LBound(schedules(i))).StartDate

            If EarliestScheduleStart = 0 Or candidate < EarliestScheduleStart Then
                EarliestScheduleStart = candidate
            End If
        End If
    Next i
End Function

Private Function FindHeaderColumn(ByRef data As Variant, ByVal headerName As String) As Long
    Dim j As Long

    For j = 1 To UBound(data, 2)
        If UCase$(Trim$(CStr(data(1, j)))) = UCase$(headerName) Then
            FindHeaderColumn = j
            Exit Function
        End If
    Next j

    Err.Raise vbObjectError + 20, , "Missing curve column: " & headerName
End Function

Private Function IsDateValue(ByVal v As Variant) As Boolean
    If IsDate(v) Then
        IsDateValue = True
    ElseIf IsNumeric(v) Then
        IsDateValue = (CDbl(v) > 20000# And CDbl(v) < 80000#)
    End If
End Function

Private Function ToDate(ByVal v As Variant) As Date
    If IsNumeric(v) Then
        ToDate = DateSerial(1899, 12, 30) + CLng(CDbl(v))
    Else
        ToDate = CDate(v)
    End If
End Function

Private Function NormalizeRate(ByVal v As Variant) As Double
    If IsError(v) Or IsEmpty(v) Or Not IsNumeric(v) Then
        Err.Raise vbObjectError + 21, , "Curve contains a blank or nonnumeric rate."
    End If

    NormalizeRate = CDbl(v)

    If Abs(NormalizeRate) > 1# Then
        NormalizeRate = NormalizeRate / 100#
    End If
End Function

Private Sub SortRatesByDate()
    Dim i As Long
    Dim j As Long
    Dim temp As RatePoint

    For i = 1 To UdfRateCount - 1
        For j = i + 1 To UdfRateCount
            If UdfRates(j).CurveDate < UdfRates(i).CurveDate Then
                temp = UdfRates(i)
                UdfRates(i) = UdfRates(j)
                UdfRates(j) = temp
            End If
        Next j
    Next i
End Sub

Private Function IsExactCurveDate(ByVal targetDate As Date) As Boolean
    Dim i As Long

    For i = 1 To UdfRateCount
        If UdfRates(i).CurveDate = targetDate Then
            IsExactCurveDate = True
            Exit Function
        End If
    Next i
End Function

Private Function PreviousCurveDate(ByVal targetDate As Date) As Date
    Dim i As Long

    For i = UdfRateCount To 1 Step -1
        If UdfRates(i).CurveDate <= targetDate Then
            PreviousCurveDate = UdfRates(i).CurveDate
            Exit Function
        End If
    Next i

    Err.Raise vbObjectError + 22, , _
        "No curve date exists on or before " & Format$(targetDate, "yyyy-mm-dd")
End Function

Private Function GetRateForTenor(ByVal rateDate As Date, ByVal tenor As String) As Double
    Dim i As Long

    For i = 1 To UdfRateCount
        If UdfRates(i).CurveDate = rateDate Then
            Select Case tenor
                Case "ON": GetRateForTenor = UdfRates(i).ONRate
                Case "1M": GetRateForTenor = UdfRates(i).M1Rate
                Case "2M": GetRateForTenor = UdfRates(i).M2Rate
                Case "3M": GetRateForTenor = UdfRates(i).M3Rate
                Case "6M": GetRateForTenor = UdfRates(i).M6Rate
                Case Else
                    Err.Raise vbObjectError + 23, , "Unsupported tenor: " & tenor
            End Select
            Exit Function
        End If
    Next i

    Err.Raise vbObjectError + 24, , _
        "Rate date not found: " & Format$(rateDate, "yyyy-mm-dd")
End Function

Private Function AddMonthsPreserveDay( _
    ByVal sourceDate As Date, _
    ByVal months As Long, _
    ByVal targetDay As Long) As Date

    Dim y As Long
    Dim m As Long
    Dim lastDay As Long

    y = Year(sourceDate)
    m = Month(sourceDate) + months

    Do While m > 12
        y = y + 1
        m = m - 12
    Loop

    Do While m < 1
        y = y - 1
        m = m + 12
    Loop

    lastDay = Day(DateSerial(y, m + 1, 0))
    AddMonthsPreserveDay = DateSerial(y, m, WorksheetFunction.Min(targetDay, lastDay))
End Function
