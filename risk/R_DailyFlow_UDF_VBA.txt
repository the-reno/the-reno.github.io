Attribute VB_Name = "RDailyFlowUDF"
Option Explicit

' Worksheet formula:
'   =r_dailyflow(last_date, years, amount, curve_range)
'
' Curve columns:
'   Date | ON | 1M | 2M | 3M | 6M
'
' Rates are plain percentage numbers, for example 3.55.
' Interest = Cash * Rate / 100 * Days / 360.
'
' Fixed-tenor roll convention:
'   Target roll dates stay anchored to the final analysis date:
'   target(n) = analysis date minus n * tenor months.
'   Actual roll date = latest curve date on or before target(n).
'   This prevents monthly schedules from drifting after weekends/holidays.
'
' Output:
'   Date, then five columns per tenor:
'   Rate | Maturity | Interest Earned | Value | Roll

Private Type RatePoint
    CurveDate As Date
    ONRate As Double
    M1Rate As Double
    M2Rate As Double
    M3Rate As Double
    M6Rate As Double
End Type

Private UdfRates() As RatePoint
Private UdfRateCount As Long

Private Const S_START As Long = 1
Private Const S_MATURITY As Long = 2
Private Const S_RATE As Long = 3
Private Const S_START_CASH As Long = 4
Private Const S_END_CASH As Long = 5

Public Function r_dailyflow( _
    ByVal last_date As Variant, _
    ByVal years As Double, _
    ByVal amount As Double, _
    ByVal curve As Range) As Variant

    On Error GoTo Fail

    Dim analysisEnd As Date
    Dim requestedStart As Date
    Dim analysisStart As Date
    Dim lookbackMonths As Long
    Dim schedules(1 To 5) As Variant
    Dim tenorNames(1 To 5) As String
    Dim result() As Variant
    Dim totalDays As Long
    Dim d As Date
    Dim dayRow As Long
    Dim tenorIndex As Long
    Dim outputCol As Long
    Dim scheduleRow As Long
    Dim isRoll As Boolean

    If Not IsDate(last_date) And Not IsNumeric(last_date) Then
        Err.Raise vbObjectError + 1, , "Invalid analysis date."
    End If

    If years <= 0 Then
        Err.Raise vbObjectError + 2, , "Lookback years must be greater than zero."
    End If

    If amount <= 0 Then
        Err.Raise vbObjectError + 3, , "Initial cash must be greater than zero."
    End If

    LoadRatesFromRange curve

    analysisEnd = ToDate(last_date)

    If CurveIndex(analysisEnd) = 0 Then
        Err.Raise vbObjectError + 4, , _
            "The analysis date must exist exactly in a complete curve row: " & _
            Format$(analysisEnd, "yyyy-mm-dd")
    End If

    lookbackMonths = CLng(Round(years * 12#, 0))
    requestedStart = DateAdd("m", -lookbackMonths, analysisEnd)

    tenorNames(1) = "ON"
    tenorNames(2) = "1M"
    tenorNames(3) = "2M"
    tenorNames(4) = "3M"
    tenorNames(5) = "6M"

    schedules(1) = BuildONSchedule(requestedStart, analysisEnd, amount)
    schedules(2) = BuildFixedSchedule(requestedStart, analysisEnd, 1, "1M", amount)
    schedules(3) = BuildFixedSchedule(requestedStart, analysisEnd, 2, "2M", amount)
    schedules(4) = BuildFixedSchedule(requestedStart, analysisEnd, 3, "3M", amount)
    schedules(5) = BuildFixedSchedule(requestedStart, analysisEnd, 6, "6M", amount)

    analysisStart = EarliestScheduleStart(schedules)

    If analysisStart = 0 Then
        Err.Raise vbObjectError + 5, , "No valid schedule exists in the requested period."
    End If

    totalDays = DateDiff("d", analysisStart, analysisEnd) + 1
    ReDim result(1 To totalDays + 1, 1 To 26)

    result(1, 1) = "Date"
    outputCol = 2

    For tenorIndex = 1 To 5
        result(1, outputCol) = tenorNames(tenorIndex) & " Rate"
        result(1, outputCol + 1) = tenorNames(tenorIndex) & " Maturity"
        result(1, outputCol + 2) = tenorNames(tenorIndex) & " Interest Earned"
        result(1, outputCol + 3) = tenorNames(tenorIndex) & " Value"
        result(1, outputCol + 4) = tenorNames(tenorIndex) & " Roll"
        outputCol = outputCol + 5
    Next tenorIndex

    dayRow = 2

    For d = analysisStart To analysisEnd
        result(dayRow, 1) = d
        outputCol = 2

        For tenorIndex = 1 To 5
            scheduleRow = FindScheduleRow(schedules(tenorIndex), d, isRoll)

            If scheduleRow > 0 Then
                result(dayRow, outputCol) = schedules(tenorIndex)(scheduleRow, S_RATE)
                result(dayRow, outputCol + 1) = schedules(tenorIndex)(scheduleRow, S_MATURITY)
                result(dayRow, outputCol + 2) = PositionInterest(schedules(tenorIndex), scheduleRow, d)
                result(dayRow, outputCol + 3) = PositionValue(schedules(tenorIndex), scheduleRow, d)
                result(dayRow, outputCol + 4) = isRoll
            Else
                result(dayRow, outputCol) = Empty
                result(dayRow, outputCol + 1) = Empty
                result(dayRow, outputCol + 2) = 0#
                result(dayRow, outputCol + 3) = 0#
                result(dayRow, outputCol + 4) = False
            End If

            outputCol = outputCol + 5
        Next tenorIndex

        dayRow = dayRow + 1
    Next d

    r_dailyflow = result
    Exit Function

Fail:
    r_dailyflow = "r_dailyflow error: " & Err.Description
End Function

Private Sub LoadRatesFromRange(ByVal curve As Range)

    Dim dataRange As Range
    Dim data As Variant
    Dim firstRow As Long
    Dim lastRow As Long
    Dim startRow As Long
    Dim i As Long
    Dim n As Long

    If curve.Columns.Count < 6 Then
        Err.Raise vbObjectError + 10, , _
            "Curve range must include Date, ON, 1M, 2M, 3M and 6M."
    End If

    firstRow = curve.Row
    lastRow = curve.Worksheet.Cells( _
        curve.Worksheet.Rows.Count, curve.Column).End(xlUp).Row

    Set dataRange = curve.Worksheet.Range( _
        curve.Worksheet.Cells(firstRow, curve.Column), _
        curve.Worksheet.Cells(lastRow, curve.Column + 5))

    data = dataRange.Value2

    If IsDateValue(data(1, 1)) Then
        startRow = 1
    Else
        startRow = 2
    End If

    ReDim UdfRates(1 To UBound(data, 1) - startRow + 1)
    n = 0

    For i = startRow To UBound(data, 1)
        If IsCompleteCurveRow(data, i) Then
            n = n + 1
            UdfRates(n).CurveDate = ToDate(data(i, 1))
            UdfRates(n).ONRate = CDbl(data(i, 2))
            UdfRates(n).M1Rate = CDbl(data(i, 3))
            UdfRates(n).M2Rate = CDbl(data(i, 4))
            UdfRates(n).M3Rate = CDbl(data(i, 5))
            UdfRates(n).M6Rate = CDbl(data(i, 6))
        End If
    Next i

    UdfRateCount = n

    If UdfRateCount = 0 Then
        Err.Raise vbObjectError + 11, , "Curve contains no complete numeric rows."
    End If

    ReDim Preserve UdfRates(1 To UdfRateCount)
    SortRatesByDate
    RemoveDuplicateDates
End Sub

Private Function IsCompleteCurveRow(ByRef data As Variant, ByVal rowNo As Long) As Boolean

    Dim j As Long

    If Len(Trim$(CStr(data(rowNo, 1)))) = 0 Then Exit Function
    If Not IsDateValue(data(rowNo, 1)) Then Exit Function

    For j = 2 To 6
        If IsError(data(rowNo, j)) Then Exit Function
        If IsEmpty(data(rowNo, j)) Then Exit Function
        If Len(Trim$(CStr(data(rowNo, j)))) = 0 Then Exit Function
        If Not IsNumeric(data(rowNo, j)) Then Exit Function
    Next j

    IsCompleteCurveRow = True
End Function

Private Sub RemoveDuplicateDates()

    Dim i As Long
    Dim n As Long

    If UdfRateCount <= 1 Then Exit Sub

    n = 1

    For i = 2 To UdfRateCount
        If UdfRates(i).CurveDate = UdfRates(n).CurveDate Then
            UdfRates(n) = UdfRates(i)
        Else
            n = n + 1
            If n <> i Then UdfRates(n) = UdfRates(i)
        End If
    Next i

    UdfRateCount = n
    ReDim Preserve UdfRates(1 To UdfRateCount)
End Sub

Private Function BuildONSchedule( _
    ByVal requestedStart As Date, _
    ByVal analysisEnd As Date, _
    ByVal amount As Double) As Variant

    Dim firstIndex As Long
    Dim lastIndex As Long
    Dim count As Long
    Dim schedule() As Variant
    Dim i As Long
    Dim rowNo As Long
    Dim startDate As Date
    Dim maturityDate As Date
    Dim cash As Double
    Dim rateUsed As Double

    firstIndex = FirstCurveIndexOnOrAfter(requestedStart)
    lastIndex = CurveIndex(analysisEnd)

    If firstIndex = 0 Or lastIndex = 0 Or firstIndex >= lastIndex Then
        BuildONSchedule = Empty
        Exit Function
    End If

    count = lastIndex - firstIndex
    ReDim schedule(1 To count, 1 To 5)

    cash = amount
    rowNo = 0

    For i = firstIndex To lastIndex - 1
        rowNo = rowNo + 1
        startDate = UdfRates(i).CurveDate
        maturityDate = UdfRates(i + 1).CurveDate
        rateUsed = UdfRates(i).ONRate

        schedule(rowNo, S_START) = startDate
        schedule(rowNo, S_MATURITY) = maturityDate
        schedule(rowNo, S_RATE) = rateUsed
        schedule(rowNo, S_START_CASH) = cash
        schedule(rowNo, S_END_CASH) = cash + _
            cash * rateUsed / 100# * _
            DateDiff("d", startDate, maturityDate) / 360#

        cash = schedule(rowNo, S_END_CASH)
    Next i

    BuildONSchedule = schedule
End Function

Private Function BuildFixedSchedule( _
    ByVal requestedStart As Date, _
    ByVal analysisEnd As Date, _
    ByVal monthsStep As Long, _
    ByVal tenor As String, _
    ByVal amount As Double) As Variant

    Const MAX_ROLLS As Long = 2000

    Dim backwardDates(1 To MAX_ROLLS) As Date
    Dim retainedDates() As Date
    Dim schedule() As Variant
    Dim targetRoll As Date
    Dim actualRoll As Date
    Dim count As Long
    Dim keepCount As Long
    Dim n As Long
    Dim i As Long
    Dim maturityDate As Date
    Dim cash As Double
    Dim rateUsed As Double

    count = 0
    n = 1

    Do
        targetRoll = DateAdd("m", -(monthsStep * n), analysisEnd)
        actualRoll = PreviousCurveDate(targetRoll)

        If actualRoll < requestedStart Then Exit Do

        If count = 0 Or actualRoll <> backwardDates(count) Then
            count = count + 1
            backwardDates(count) = actualRoll
        End If

        n = n + 1
        If count >= MAX_ROLLS Then Exit Do
    Loop

    If count = 0 Then
        BuildFixedSchedule = Empty
        Exit Function
    End If

    keepCount = count
    ReDim retainedDates(1 To keepCount)

    For i = 1 To keepCount
        retainedDates(i) = backwardDates(keepCount - i + 1)
    Next i

    ReDim schedule(1 To keepCount, 1 To 5)
    cash = amount

    For i = 1 To keepCount
        If i < keepCount Then
            maturityDate = retainedDates(i + 1)
        Else
            maturityDate = analysisEnd
        End If

        rateUsed = GetRateForTenor(retainedDates(i), tenor)

        schedule(i, S_START) = retainedDates(i)
        schedule(i, S_MATURITY) = maturityDate
        schedule(i, S_RATE) = rateUsed
        schedule(i, S_START_CASH) = cash
        schedule(i, S_END_CASH) = cash + _
            cash * rateUsed / 100# * _
            DateDiff("d", retainedDates(i), maturityDate) / 360#

        cash = schedule(i, S_END_CASH)
    Next i

    BuildFixedSchedule = schedule
End Function

Private Function FindScheduleRow( _
    ByVal schedule As Variant, _
    ByVal calendarDate As Date, _
    ByRef isRoll As Boolean) As Long

    Dim i As Long
    Dim firstRow As Long
    Dim lastRow As Long

    FindScheduleRow = 0
    isRoll = False

    If IsEmpty(schedule) Then Exit Function

    firstRow = LBound(schedule, 1)
    lastRow = UBound(schedule, 1)

    For i = firstRow To lastRow
        If calendarDate = CDate(schedule(i, S_START)) Then
            FindScheduleRow = i
            isRoll = True
            Exit Function
        End If
    Next i

    For i = firstRow To lastRow
        If calendarDate > CDate(schedule(i, S_START)) And _
           calendarDate < CDate(schedule(i, S_MATURITY)) Then
            FindScheduleRow = i
            Exit Function
        End If
    Next i

    If calendarDate = CDate(schedule(lastRow, S_MATURITY)) Then
        FindScheduleRow = lastRow
        isRoll = False
    End If
End Function

Private Function PositionInterest( _
    ByVal schedule As Variant, _
    ByVal scheduleRow As Long, _
    ByVal calendarDate As Date) As Double

    Dim startDate As Date
    Dim maturityDate As Date
    Dim accrualEnd As Date
    Dim startCash As Double
    Dim rateUsed As Double

    startDate = CDate(schedule(scheduleRow, S_START))
    maturityDate = CDate(schedule(scheduleRow, S_MATURITY))
    startCash = CDbl(schedule(scheduleRow, S_START_CASH))
    rateUsed = CDbl(schedule(scheduleRow, S_RATE))

    If calendarDate <= startDate Then
        PositionInterest = 0#
        Exit Function
    End If

    accrualEnd = calendarDate
    If accrualEnd > maturityDate Then accrualEnd = maturityDate

    PositionInterest = startCash * rateUsed / 100# * _
        DateDiff("d", startDate, accrualEnd) / 360#
End Function

Private Function PositionValue( _
    ByVal schedule As Variant, _
    ByVal scheduleRow As Long, _
    ByVal calendarDate As Date) As Double

    If calendarDate >= CDate(schedule(scheduleRow, S_MATURITY)) Then
        PositionValue = CDbl(schedule(scheduleRow, S_END_CASH))
    Else
        PositionValue = CDbl(schedule(scheduleRow, S_START_CASH)) + _
            PositionInterest(schedule, scheduleRow, calendarDate)
    End If
End Function

Private Function EarliestScheduleStart(ByRef schedules() As Variant) As Date

    Dim i As Long
    Dim candidate As Date

    EarliestScheduleStart = 0

    For i = LBound(schedules) To UBound(schedules)
        If Not IsEmpty(schedules(i)) Then
            candidate = CDate(schedules(i)(LBound(schedules(i), 1), S_START))

            If EarliestScheduleStart = 0 Or candidate < EarliestScheduleStart Then
                EarliestScheduleStart = candidate
            End If
        End If
    Next i
End Function

Private Function FirstCurveIndexOnOrAfter(ByVal targetDate As Date) As Long

    Dim i As Long

    For i = 1 To UdfRateCount
        If UdfRates(i).CurveDate >= targetDate Then
            FirstCurveIndexOnOrAfter = i
            Exit Function
        End If
    Next i
End Function

Private Function CurveIndex(ByVal targetDate As Date) As Long

    Dim i As Long

    For i = 1 To UdfRateCount
        If UdfRates(i).CurveDate = targetDate Then
            CurveIndex = i
            Exit Function
        End If
    Next i
End Function

Private Function PreviousCurveDate(ByVal targetDate As Date) As Date

    Dim i As Long

    For i = UdfRateCount To 1 Step -1
        If UdfRates(i).CurveDate <= targetDate Then
            PreviousCurveDate = UdfRates(i).CurveDate
            Exit Function
        End If
    Next i

    Err.Raise vbObjectError + 20, , _
        "No curve date exists on or before " & Format$(targetDate, "yyyy-mm-dd")
End Function

Private Function GetRateForTenor( _
    ByVal rateDate As Date, _
    ByVal tenor As String) As Double

    Dim i As Long

    i = CurveIndex(rateDate)

    If i = 0 Then
        Err.Raise vbObjectError + 21, , _
            "Rate date not found: " & Format$(rateDate, "yyyy-mm-dd")
    End If

    Select Case tenor
        Case "ON": GetRateForTenor = UdfRates(i).ONRate
        Case "1M": GetRateForTenor = UdfRates(i).M1Rate
        Case "2M": GetRateForTenor = UdfRates(i).M2Rate
        Case "3M": GetRateForTenor = UdfRates(i).M3Rate
        Case "6M": GetRateForTenor = UdfRates(i).M6Rate
        Case Else
            Err.Raise vbObjectError + 22, , "Unsupported tenor: " & tenor
    End Select
End Function

Private Function IsDateValue(ByVal v As Variant) As Boolean

    If IsDate(v) Then
        IsDateValue = True
    ElseIf IsNumeric(v) Then
        IsDateValue = (CDbl(v) > 20000# And CDbl(v) < 80000#)
    End If
End Function

Private Function ToDate(ByVal v As Variant) As Date

    If IsNumeric(v) Then
        ToDate = DateSerial(1899, 12, 30) + Int(CDbl(v))
    Else
        ToDate = DateValue(CDate(v))
    End If
End Function

Private Sub SortRatesByDate()

    Dim i As Long
    Dim j As Long
    Dim temp As RatePoint

    For i = 1 To UdfRateCount - 1
        For j = i + 1 To UdfRateCount
            If UdfRates(j).CurveDate < UdfRates(i).CurveDate Then
                temp = UdfRates(i)
                UdfRates(i) = UdfRates(j)
                UdfRates(j) = temp
            End If
        Next j
    Next i
End Sub
